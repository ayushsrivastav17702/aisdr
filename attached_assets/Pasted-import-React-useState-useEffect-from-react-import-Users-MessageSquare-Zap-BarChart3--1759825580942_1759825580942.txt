import React, { useState, useEffect } from 'react';
import { Users, MessageSquare, Zap, BarChart3, Settings, Plus, RefreshCw, Sparkles, X, Upload, CheckCircle, Trash2 } from 'lucide-react';

// ============================================
// MAIN SEQUENCE BUILDER WITH ALL TABS
// ============================================

export function CompleteSequenceBuilder({ sequenceId = 1 }) {
  const [activeTab, setActiveTab] = useState('sequence');
  const [sequence, setSequence] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadSequence();
  }, [sequenceId]);

  const loadSequence = async () => {
    try {
      const response = await fetch(`/api/sequences/${sequenceId}`);
      const data = await response.json();
      setSequence(data);
    } catch (error) {
      console.error('Error loading sequence:', error);
    } finally {
      setLoading(false);
    }
  };

  const tabs = [
    { id: 'sequence', label: 'Sequence', icon: null },
    { id: 'prospects', label: 'Prospects', icon: Users },
    { id: 'replies', label: 'Replies', icon: MessageSquare },
    { id: 'ai-followup', label: 'AI Follow-up', icon: Zap },
    { id: 'tracking', label: 'Email Tracking', icon: BarChart3 },
    { id: 'settings', label: 'Settings', icon: Settings }
  ];

  if (loading) return <div className="p-8">Loading...</div>;

  return (
    <div className="min-h-screen bg-gray-50">
      <div className="max-w-7xl mx-auto p-6">
        <div className="bg-white rounded-lg shadow-sm p-6 mb-6">
          <h1 className="text-3xl font-bold text-gray-900">{sequence?.name || 'Sequence Builder'}</h1>
        </div>

        <div className="bg-white rounded-lg shadow-sm mb-6">
          <div className="border-b border-gray-200 flex overflow-x-auto">
            {tabs.map((tab) => {
              const Icon = tab.icon;
              return (
                <button
                  key={tab.id}
                  onClick={() => setActiveTab(tab.id)}
                  className={`flex items-center gap-2 px-6 py-4 text-sm font-medium border-b-2 whitespace-nowrap ${
                    activeTab === tab.id
                      ? 'border-blue-600 text-blue-600'
                      : 'border-transparent text-gray-600 hover:text-gray-900'
                  }`}
                >
                  {Icon && <Icon className="w-4 h-4" />}
                  {tab.label}
                </button>
              );
            })}
          </div>
        </div>

        <div className="bg-white rounded-lg shadow-sm p-6">
          {activeTab === 'sequence' && <SequenceTab sequenceId={sequenceId} />}
          {activeTab === 'prospects' && <ProspectsTab sequenceId={sequenceId} />}
          {activeTab === 'replies' && <RepliesTab sequenceId={sequenceId} />}
          {activeTab === 'ai-followup' && <AIFollowupTab sequenceId={sequenceId} />}
          {activeTab === 'tracking' && <EmailTrackingTab sequenceId={sequenceId} />}
          {activeTab === 'settings' && <SettingsTab sequenceId={sequenceId} />}
        </div>
      </div>
    </div>
  );
}

// ============================================
// REPLIES TAB WITH REAL-TIME
// ============================================

function RepliesTab({ sequenceId }) {
  const [replies, setReplies] = useState([]);
  const [stats, setStats] = useState({ total: 0, responseRate: 0, latestReply: null });
  const [autoRefresh, setAutoRefresh] = useState(true);

  useEffect(() => {
    loadReplies();
    if (autoRefresh) {
      const interval = setInterval(loadReplies, 30000); // 30 seconds
      return () => clearInterval(interval);
    }
  }, [autoRefresh, sequenceId]);

  const loadReplies = async () => {
    try {
      const response = await fetch(`/api/sequences/${sequenceId}/replies`);
      const data = await response.json();
      setReplies(data.replies || []);
      setStats({
        total: data.total || 0,
        responseRate: data.responseRate || 0,
        latestReply: data.latestReply
      });
    } catch (error) {
      console.error('Error loading replies:', error);
    }
  };

  return (
    <div>
      <div className="flex items-center justify-between mb-6">
        <div>
          <h2 className="text-2xl font-bold text-gray-900">Email Replies</h2>
          <p className="text-gray-600">{stats.total} prospects replied</p>
        </div>
        <div className="flex items-center gap-3">
          <label className="flex items-center gap-2 text-sm">
            <input
              type="checkbox"
              checked={autoRefresh}
              onChange={(e) => setAutoRefresh(e.target.checked)}
              className="rounded"
            />
            Auto-refresh (30s)
          </label>
          <button
            onClick={loadReplies}
            className="px-4 py-2 border border-gray-300 rounded-lg hover:bg-gray-50 flex items-center gap-2"
          >
            <RefreshCw className="w-4 h-4" />
            Refresh
          </button>
        </div>
      </div>

      <div className="grid grid-cols-3 gap-6 mb-6">
        <div className="bg-blue-50 rounded-lg p-6">
          <div className="text-sm text-blue-600 mb-1">Total Replies</div>
          <div className="text-3xl font-bold text-blue-900">{stats.total}</div>
        </div>
        <div className="bg-green-50 rounded-lg p-6">
          <div className="text-sm text-green-600 mb-1">Response Rate</div>
          <div className="text-3xl font-bold text-green-900">{stats.responseRate}%</div>
        </div>
        <div className="bg-purple-50 rounded-lg p-6">
          <div className="text-sm text-purple-600 mb-1">Latest Reply</div>
          <div className="text-lg font-semibold text-purple-900">{stats.latestReply || 'None'}</div>
        </div>
      </div>

      {replies.length === 0 ? (
        <div className="text-center py-16 border-2 border-dashed border-gray-300 rounded-lg">
          <MessageSquare className="w-16 h-16 text-gray-300 mx-auto mb-4" />
          <h3 className="text-xl font-semibold text-gray-900 mb-2">No Replies Yet</h3>
          <p className="text-gray-600">Replies will appear here in real-time</p>
          {autoRefresh && <p className="text-green-600 text-sm mt-4">✓ Auto-refresh enabled</p>}
        </div>
      ) : (
        <div className="space-y-4">
          {replies.map((reply) => (
            <div key={reply.id} className="border rounded-lg p-4 hover:shadow-md transition">
              <div className="flex items-start justify-between mb-3">
                <div className="flex items-center gap-3">
                  <div className="w-10 h-10 bg-blue-100 rounded-full flex items-center justify-center">
                    <span className="text-blue-600 font-semibold">
                      {reply.prospect?.firstName?.[0]}{reply.prospect?.lastName?.[0]}
                    </span>
                  </div>
                  <div>
                    <div className="font-semibold">{reply.prospect?.firstName} {reply.prospect?.lastName}</div>
                    <div className="text-sm text-gray-600">{reply.prospect?.company}</div>
                  </div>
                </div>
                <div className="text-sm text-gray-500">{new Date(reply.receivedAt).toLocaleString()}</div>
              </div>
              <div className="bg-gray-50 rounded p-3 text-gray-700 mb-2">
                {reply.replyContent}
              </div>
              {reply.aiSummary && (
                <div className="text-sm text-gray-600">
                  <strong>AI Summary:</strong> {reply.aiSummary}
                </div>
              )}
              {reply.sentiment && (
                <span className={`inline-block px-2 py-1 rounded text-xs mt-2 ${
                  reply.sentiment === 'positive' ? 'bg-green-100 text-green-700' :
                  reply.sentiment === 'negative' ? 'bg-red-100 text-red-700' :
                  'bg-gray-100 text-gray-700'
                }`}>
                  {reply.sentiment}
                </span>
              )}
            </div>
          ))}
        </div>
      )}
    </div>
  );
}

// ============================================
// AI FOLLOW-UP TAB
// ============================================

function AIFollowupTab({ sequenceId }) {
  const [active, setActive] = useState(false);
  const [config, setConfig] = useState({
    daysBetween: 3,
    maxFollowups: 3,
    followupType: 'gentle',
    triggerCondition: 'no_response'
  });

  const saveConfig = async () => {
    try {
      await fetch(`/api/sequences/${sequenceId}/ai-followup`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(config)
      });
      alert('Configuration saved!');
    } catch (error) {
      alert('Failed to save configuration');
    }
  };

  return (
    <div>
      <h2 className="text-2xl font-bold mb-6">AI Follow-up Configuration</h2>

      <div className="bg-gradient-to-r from-purple-50 to-blue-50 rounded-lg p-6 mb-6">
        <div className="flex items-center justify-between">
          <div>
            <h3 className="text-lg font-semibold mb-2">AI Follow-up Scheduler</h3>
            <p className="text-gray-600">Automatically sends follow-ups based on conditions</p>
          </div>
          <button
            onClick={() => setActive(!active)}
            className={`px-6 py-3 rounded-lg font-medium ${
              active ? 'bg-green-600 text-white' : 'bg-blue-600 text-white'
            }`}
          >
            {active ? 'Scheduler Active' : 'Start Scheduler'}
          </button>
        </div>
      </div>

      <div className="space-y-6">
        <div>
          <label className="block text-sm font-medium mb-2">Days Between Follow-ups</label>
          <input
            type="number"
            value={config.daysBetween}
            onChange={(e) => setConfig({ ...config, daysBetween: parseInt(e.target.value) })}
            className="w-full px-4 py-2 border rounded-lg"
            min="1"
            max="30"
          />
        </div>

        <div>
          <label className="block text-sm font-medium mb-2">Maximum Follow-ups</label>
          <input
            type="number"
            value={config.maxFollowups}
            onChange={(e) => setConfig({ ...config, maxFollowups: parseInt(e.target.value) })}
            className="w-full px-4 py-2 border rounded-lg"
            min="1"
            max="10"
          />
        </div>

        <div>
          <label className="block text-sm font-medium mb-2">Follow-up Type</label>
          <select
            value={config.followupType}
            onChange={(e) => setConfig({ ...config, followupType: e.target.value })}
            className="w-full px-4 py-2 border rounded-lg"
          >
            <option value="gentle">Gentle Reminder</option>
            <option value="value">Value Add</option>
            <option value="urgent">Urgent</option>
            <option value="breakup">Breakup Email</option>

        </button>
      </div>

// ============================================
// LINKEDIN DATA INPUT MODAL
// ============================================

function LinkedInDataInputModal({ open, onClose, prospectId, prospectName, onPersonalizationComplete }) {
  const [step, setStep] = useState(1);
  const [linkedInData, setLinkedInData] = useState({
    profileText: '',
    headline: '',
    recentPosts: [''],
    recentComments: [''],
    skills: []
  });
  const [result, setResult] = useState(null);
  const [processing, setProcessing] = useState(false);

  if (!open) return null;

  const handleGenerate = async () => {
    setProcessing(true);
    setStep(2);

    try {
      const response = await fetch('/api/personalization/manual-linkedin', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          prospectId,
          linkedInData
        })
      });

      const data = await response.json();
      setResult(data);
      setStep(3);
      onPersonalizationComplete?.(data);
    } catch (error) {
      alert('Failed to generate personalization');
      setStep(1);
    } finally {
      setProcessing(false);
    }
  };

  return (
    <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50 p-4">
      <div className="bg-white rounded-xl max-w-4xl w-full max-h-[90vh] overflow-y-auto">
        <div className="sticky top-0 bg-white border-b p-6 flex items-center justify-between">
          <div>
            <h2 className="text-2xl font-bold">AI Personalization with LinkedIn Data</h2>
            <p className="text-gray-600 mt-1">Prospect: {prospectName}</p>
          </div>
          <button onClick={onClose}><X className="w-6 h-6" /></button>
        </div>

        <div className="p-6">
          {step === 1 && (
            <div className="space-y-6">
              <div className="bg-blue-50 border border-blue-200 rounded-lg p-4">
                <div className="flex items-start gap-3">
                  <Sparkles className="w-5 h-5 text-blue-600 mt-0.5" />
                  <div>
                    <h3 className="font-semibold text-blue-900 mb-1">How it works</h3>
                    <p className="text-sm text-blue-800">
                      Paste LinkedIn data. OpenAI analyzes it to generate highly personalized emails (85-95% score).
                    </p>
                  </div>
                </div>
              </div>

              <div>
                <label className="block text-sm font-medium mb-2">Profile/About Section</label>
                <textarea
                  value={linkedInData.profileText}
                  onChange={(e) => setLinkedInData({ ...linkedInData, profileText: e.target.value })}
                  placeholder="Paste their LinkedIn About section..."
                  rows="4"
                  className="w-full px-4 py-3 border rounded-lg"
                />
              </div>

              <div>
                <label className="block text-sm font-medium mb-2">Recent Posts</label>
                {linkedInData.recentPosts.map((post, i) => (
                  <textarea
                    key={i}
                    value={post}
                    onChange={(e) => {
                      const newPosts = [...linkedInData.recentPosts];
                      newPosts[i] = e.target.value;
                      setLinkedInData({ ...linkedInData, recentPosts: newPosts });
                    }}
                    placeholder={`Paste recent post ${i + 1}...`}
                    rows="3"
                    className="w-full px-4 py-3 border rounded-lg mb-2"
                  />
                ))}
              </div>

              <div className="flex gap-3 pt-4 border-t">
                <button onClick={onClose} className="px-6 py-3 border rounded-lg">Cancel</button>
                <button
                  onClick={handleGenerate}
                  disabled={!linkedInData.profileText && !linkedInData.recentPosts[0]}
                  className="flex-1 px-6 py-3 bg-gradient-to-r from-blue-600 to-purple-600 text-white rounded-lg disabled:opacity-50"
                >
                  <Sparkles className="w-5 h-5 inline mr-2" />
                  Generate AI Personalization
                </button>
              </div>
            </div>
          )}

          {step === 2 && (
            <div className="text-center py-12">
              <div className="inline-block animate-spin rounded-full h-16 w-16 border-4 border-blue-600 border-t-transparent mb-4"></div>
              <h3 className="text-xl font-semibold mb-2">Analyzing with OpenAI...</h3>
            </div>
          )}

          {step === 3 && result && (
            <div className="space-y-6">
              <div className="bg-green-50 border border-green-200 rounded-lg p-4">
                <CheckCircle className="w-5 h-5 text-green-600 inline mr-2" />
                Personalization Complete! Score: {result.personalizationScore}%
              </div>

              <div>
                <h3 className="text-lg font-semibold mb-3">Generated Email</h3>
                <div className="border rounded-lg p-4 space-y-3">
                  <div>
                    <div className="text-sm font-medium text-gray-700 mb-1">Subject</div>
                    <div className="text-gray-900">{result.email?.subject}</div>
                  </div>
                  <div>
                    <div className="text-sm font-medium text-gray-700 mb-1">Body</div>
                    <div className="text-gray-900 whitespace-pre-wrap">{result.email?.body}</div>
                  </div>
                </div>
              </div>

              <button onClick={onClose} className="w-full px-6 py-3 bg-blue-600 text-white rounded-lg">
                Done
              </button>
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
