javascriptconst Prospect = require('../models/Prospect');
const ProspectSearchService = require('../services/prospectSearchService');
const { logAudit } = require('../utils/auditLog');

exports.searchProspects = async (req, res) => {
    try {
        const { industry, companySize, jobTitles, location, limit } = req.body;

        // Validate
        if (!industry || !jobTitles || jobTitles.length === 0) {
            return res.status(400).json({ error: 'Industry and job titles are required' });
        }

        const searchService = new ProspectSearchService(req.user.tenantId, req.user.userId);
        
        const searchResult = await searchService.search({
            industry,
            companySize,
            jobTitles,
            location,
            limit: limit || 50
        });

        await logAudit({
            tenantId: req.user.tenantId,
            userId: req.user.userId,
            action: 'prospect_search',
            entityType: 'prospect',
            details: { provider: searchResult.provider, count: searchResult.results.length }
        });

        res.json({
            provider: searchResult.provider,
            prospects: searchResult.results,
            cost: searchResult.cost,
            count: searchResult.results.length
        });

    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.saveProspects = async (req, res) => {
    try {
        const { prospects } = req.body;

        if (!prospects || prospects.length === 0) {
            return res.status(400).json({ error: 'No prospects to save' });
        }

        const saved = await Prospect.bulkCreate(
            prospects,
            req.user.tenantId,
            req.user.userId
        );

        await logAudit({
            tenantId: req.user.tenantId,
            userId: req.user.userId,
            action: 'prospects_imported',
            entityType: 'prospect',
            details: { count: saved.length }
        });

        res.json({ 
            message: `${saved.length} prospects saved successfully`,
            duplicates: prospects.length - saved.length,
            prospects: saved
        });

    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.getProspects = async (req, res) => {
    try {
        const { status, search } = req.query;
        const prospects = await Prospect.findByTenant(req.user.tenantId, { status, search });
        res.json({ prospects });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.getProspectById = async (req, res) => {
    try {
        const prospect = await Prospect.findById(req.params.id, req.user.tenantId);
        if (!prospect) {
            return res.status(404).json({ error: 'Prospect not found' });
        }
        res.json({ prospect });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.updateProspect = async (req, res) => {
    try {
        const prospect = await Prospect.update(req.params.id, req.user.tenantId, req.body);
        if (!prospect) {
            return res.status(404).json({ error: 'Prospect not found' });
        }
        res.json({ prospect });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.deleteProspect = async (req, res) => {
    try {
        await Prospect.delete(req.params.id, req.user.tenantId);
        res.json({ message: 'Prospect deleted successfully' });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.getStats = async (req, res) => {
    try {
        const stats = await Prospect.getStats(req.user.tenantId);
        res.json({ stats });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.importCSV = async (req, res) => {
    try {
        const { prospects } = req.body;
        
        const saved = await Prospect.bulkCreate(
            prospects,
            req.user.tenantId,
            req.user.userId
        );

        res.json({ 
            message: `${saved.length} prospects imported`,
            duplicates: prospects.length - saved.length
        });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

Step 5.7: Frontend - AI Prospecting Page
File: frontend/src/pages/User/AIProspecting.jsx
jsximport { useState } from 'react';
import api from '../../services/api';

export default function AIProspecting() {
    const [step, setStep] = useState(1);
    const [icpCriteria, setIcpCriteria] = useState({
        industry: '',
        companySize: '51-200',
        jobTitles: [''],
        location: '',
        limit: 50
    });
    const [searchResults, setSearchResults] = useState(null);
    const [loading, setLoading] = useState(false);
    const [selectedProspects, setSelectedProspects] = useState([]);

    const handleSearch = async () => {
        setLoading(true);
        try {
            const response = await api.post('/prospects/search', icpCriteria);
            setSearchResults(response.data);
            setStep(2);
        } catch (error) {
            alert(error.response?.data?.error || 'Search failed');
        } finally {
            setLoading(false);
        }
    };

    const handleSave = async () => {
        if (selectedProspects.length === 0) {
            alert('Please select at least one prospect');
            return;
        }

        setLoading(true);
        try {
            const prospectsToSave = searchResults.prospects.filter((_, idx) => 
                selectedProspects.includes(idx)
            );
            
            await api.post('/prospects/save', { prospects: prospectsToSave });
            alert(`${prospectsToSave.length} prospects saved successfully!`);
            setStep(1);
            setSearchResults(null);
            setSelectedProspects([]);
        } catch (error) {
            alert('Failed to save prospects');
        } finally {
            setLoading(false);
        }
    };

    const toggleSelect = (idx) => {
        if (selectedProspects.includes(idx)) {
            setSelectedProspects(selectedProspects.filter(i => i !== idx));
        } else {
            setSelectedProspects([...selectedProspects, idx]);
        }
    };

    const toggleSelectAll = () => {
        if (selectedProspects.length === searchResults.prospects.length) {
            setSelectedProspects([]);
        } else {
            setSelectedProspects(searchResults.prospects.map((_, idx) => idx));
        }
    };

    const addJobTitle = () => {
        setIcpCriteria({
            ...icpCriteria, 
            jobTitles: [...icpCriteria.jobTitles, '']
        });
    };

    const updateJobTitle = (idx, value) => {
        const newTitles = [...icpCriteria.jobTitles];
        newTitles[idx] = value;
        setIcpCriteria({...icpCriteria, jobTitles: newTitles});
    };

    const removeJobTitle = (idx) => {
        if (icpCriteria.jobTitles.length > 1) {
            const newTitles = icpCriteria.jobTitles.filter((_, i) => i !== idx);
            setIcpCriteria({...icpCriteria, jobTitles: newTitles});
        }
    };

    return (
        <div className="p-8 max-w-6xl mx-auto">
            <h1 className="text-3xl font-bold mb-8">🔍 AI Prospect Finder</h1>

            {step === 1 && (
                <div className="bg-white p-6 rounded-lg shadow">
                    <h2 className="text-xl font-bold mb-4">Define Your Ideal Customer Profile</h2>

                    <div className="space-y-4">
                        <div>
                            <label className="block text-sm font-medium mb-2">Industry *</label>
                            <input
                                type="text"
                                value={icpCriteria.industry}
                                onChange={(e) => setIcpCriteria({...icpCriteria, industry: e.target.value})}
                                className="w-full border rounded px-3 py-2"
                                placeholder="e.g., Technology, Healthcare, Finance"
                                required
                            />
                        </div>

                        <div>
                            <label className="block text-sm font-medium mb-2">Company Size</label>
                            <select
                                value={icpCriteria.companySize}
                                onChange={(e) => setIcpCriteria({...icpCriteria, companySize: e.target.value})}
                                className="w-full border rounded px-3 py-2"
                            >
                                <option value="1-10">1-10 employees</option>
                                <option value="11-50">11-50 employees</option>
                                <option value="51-200">51-200 employees</option>
                                <option value="201-500">201-500 employees</option>
                                <option value="501-1000">501-1000 employees</option>
                                <option value="1001-5000">1001-5000 employees</option>
                                <option value="5001+">5001+ employees</option>
                            </select>
                        </div>

                        <div>
                            <label className="block text-sm font-medium mb-2">Job Titles *</label>
                            {icpCriteria.jobTitles.map((title, idx) => (
                                <div key={idx} className="flex gap-2 mb-2">
                                    <input
                                        type="text"
                                        value={title}
                                        onChange={(e) => updateJobTitle(idx, e.target.value)}
                                        className="flex-1 border rounded px-3 py-2"
                                        placeholder="e.g., VP of Sales, Head of Marketing"
                                    />
                                    {icpCriteria.jobTitles.length > 1 && (
                                        <button
                                            type="button"
                                            onClick={() => removeJobTitle(idx)}
                                            className="px-3 py-2 bg-red-100 text-red-600 rounded hover:bg-red-200"
                                        >
                                            Remove
                                        </button>
                                    )}
                                </div>
                            ))}
                            <button
                                type="button"
                                onClick={addJobTitle}
                                className="text-blue-600 hover:underline text-sm"
                            >
                                + Add Another Job Title
                            </button>
                        </div>

                        <div>
                            <label className="block text-sm font-medium mb-2">Location (Optional)</label>
                            <input
                                type="text"
                                value={icpCriteria.location}
                                onChange={(e) => setIcpCriteria({...icpCriteria, location: e.target.value})}
                                className="w-full border rounded px-3 py-2"
                                placeholder="e.g., United States, California"
                            />
                        </div>

                        <div>
                            <label className="block text-sm font-medium mb-2">Number of Prospects</label>
                            <input
                                type="number"
                                value={icpCriteria.limit}
                                onChange={(e) => setIcpCriteria({...icpCriteria, limit: parseInt(e.target.value)})}
                                className="w-full border rounded px-3 py-2"
                                min="10"
                                max="500"
                            />
                        </div>
                    </div>

                    <div className="mt-6 bg-blue-50 border border-blue-200 p-4 rounded">
                        <p className="text-sm text-blue-800">
                            <strong>🤖 How it works:</strong> We search <strong>Perplexity</strong> → <strong>Apollo</strong> → <strong>Lusha</strong> → <strong>OpenRouter</strong> to find prospects. You only pay for what we find.
                        </p>
                    </div>

                    <button
                        onClick={handleSearch}
                        disabled={loading || !icpCriteria.industry || icpCriteria.jobTitles.some(t => !t)}
                        className="w-full mt-6 bg-blue-600 text-white py-3 rounded-lg hover:bg-blue-700 disabled:opacity-50 font-medium"
                    >
                        {loading ? '🔍 Searching...' : '🚀 Find Prospects'}
                    </button>
                </div>
            )}

            {step === 2 && searchResults && (
                <div className="bg-white p-6 rounded-lg shadow">
                    <div className="flex justify-between items-center mb-6">
                        <div>
                            <h2 className="text-xl font-bold">✨ Search Results</h2>
                            <p className="text-sm text-gray-600">
                                Found <strong>{searchResults.count}</strong> prospects using <strong className="text-blue-600">{searchResults.provider}</strong>
                                {searchResults.cost > 0 && (
                                    <span className="ml-2">• Cost: <strong>${searchResults.cost.toFixed(4)}</strong></span>
                                )}
                            </p>
                        </div>
                        <button
                            onClick={() => { setStep(1); setSearchResults(null); setSelectedProspects([]); }}
                            className="text-gray-600 hover:underline"
                        >
                            ← New Search
                        </button>
                    </div>

                    <div className="mb-4 flex justify-between items-center border-b pb-4">
                        <div>
                            <button
                                onClick={toggleSelectAll}
                                className="text-blue-600 hover:underline text-sm font-medium"
                            >
                                {selectedProspects.length === searchResults.prospects.length 
                                    ? '☐ Deselect All' 
                                    : '☑ Select All'}
                            </button>
                            <span className="ml-4 text-sm text-gray-600">
                                <strong>{selectedProspects.length}</strong> selected
                            </span>
                        </div>
                        <button
                            onClick={handleSave}
                            disabled={loading || selectedProspects.length === 0}
                            className="bg-green-600 text-white px-6 py-2 rounded-lg hover:bg-green-700 disabled:opacity-50 font-medium"
                        >
                            {loading ? 'Saving...' : `💾 Save ${selectedProspects.length} Prospects`}
                        </button>
                    </div>

                    <div className="border rounded overflow-auto max-h-[600px]">
                        <table className="w-full">
                            <thead className="bg-gray-50 sticky top-0">
                                <tr>
                                    <th className="text-left p-3 w-12"></th>
                                    <th className="text-left p-3">Name</th>
                                    <th className="text-left p-3">Title</th>
                                    <th className="text-left p-3">Company</th>
                                    <th className="text-left p-3">Email</th>
                                    <th className="text-left p-3">Location</th>
                                </tr>
                            </thead>
                            <tbody>
                                {searchResults.prospects.map((prospect, idx) => (
                                    <tr key={idx} className="border-t hover:bg-gray-50">
                                        <td className="p-3">
                                            <input
                                                type="checkbox"
                                                checked={selectedProspects.includes(idx)}
                                                onChange={() => toggleSelect(idx)}
                                                className="w-4 h-4 cursor-pointer"
                                            />
                                        </td>
                                        <td className="p-3 font-medium">
                                            {prospect.fullName || `${prospect.firstName} ${prospect.lastName}`}
                                        </td>
                                        <td className="p-3 text-sm">{prospect.jobTitle}</td>
                                        <td className="p-3 text-sm">{prospect.companyName}</td>
                                        <td className="p-3 text-sm">{prospect.email}</td>
                                        <td className="p-3 text-sm">{prospect.location || '-'}</td>
                                    </tr>
                                ))}
                            </tbody>
                        </table>
                    </div>

                    {searchResults.provider === 'openrouter' && (
                        <div className="mt-4 bg-yellow-50 border border-yellow-200 p-3 rounded text-sm">
                            ⚠️ <strong>Note:</strong> These prospects were AI-generated as a fallback. Please verify contact information before use.
                        </div>
                    )}
                </div>
            )}
        </div>
    );
}
