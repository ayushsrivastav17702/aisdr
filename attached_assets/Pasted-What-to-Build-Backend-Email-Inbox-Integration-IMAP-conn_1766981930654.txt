Database Tables:
sqlCREATE TABLE prospect_replies (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    prospect_id UUID REFERENCES prospects(id),
    campaign_id UUID REFERENCES campaigns(id),
    message_id UUID REFERENCES messages(id),
    subject VARCHAR(255),
    body TEXT,
    sentiment VARCHAR(50), -- positive, neutral, negative, ooo
    intent VARCHAR(50), -- meeting_request, question, objection, unsubscribe
    priority INTEGER DEFAULT 5,
    read BOOLEAN DEFAULT false,
    replied BOOLEAN DEFAULT false,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE conversations (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    prospect_id UUID REFERENCES prospects(id),
    subject VARCHAR(255),
    last_message_at TIMESTAMP,
    message_count INTEGER DEFAULT 0,
    status VARCHAR(50), -- active, archived, qualified
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

sqlCREATE TABLE metrics_daily (

    tenant_id UUID REFERENCES tenants(id),
    user_id UUID REFERENCES users(id),
    date DATE NOT NULL,
    emails_sent INTEGER DEFAULT 0,

CREATE TABLE user_quotas (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    user_id UUID REFERENCES users(id),
    period VARCHAR(50), -- daily, weekly, monthly, quarterly
    quota_type VARCHAR(50), -- meetings, replies, emails
    quota_value INTEGER NOT NULL,
    current_value INTEGER DEFAULT 0,
    start_date DATE,
    end_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
