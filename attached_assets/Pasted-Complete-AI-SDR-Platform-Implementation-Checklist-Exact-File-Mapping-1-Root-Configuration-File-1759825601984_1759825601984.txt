📊 Quick Reference: API Endpoints
# Sequences
GET    /api/sequences                    List all sequences
POST   /api/sequences                    Create sequence
GET    /api/sequences/:id                Get sequence
POST   /api/sequences/:id/steps          Add step
GET    /api/sequences/:id/prospects      List prospects
POST   /api/sequences/:id/prospects      Add prospects
GET    /api/sequences/:id/tracking       Get tracking stats

# Personalization
POST   /api/personalization/manual-linkedin      Single personalization
POST   /api/personalization/manual-linkedin-batch Batch personalization
POST   /api/personalization/preview              Preview analysis

# Webhooks
POST   /webhooks/email-reply              Record reply
POST   /webhooks/email-opened             Record open
POST   /webhooks/email-clicked            Record click
GET    /webhooks/pixel/:trackingId        Tracking pixel

# Prospects
GET    /api/prospects                     List all prospects
POST   /api/prospects                     Create prospect

Issue 1: Database Connection Failed
bash# Check DATABASE_URL
echo $DATABASE_URL

# Test connection
psql $DATABASE_URL

# Solution: Update .env with correct URL
Issue 2: OpenAI API Error
bash# Verify API key
echo $OPENAI_API_KEY

# Test key
curl https://api.openai.com/v1/models \
  -H "Authorization: Bearer $OPENAI_API_KEY"

# Solution: Get valid key from OpenAI platform
Issue 3: CORS Error
bash# Add to backend/src/index.ts
app.use(cors({
  origin: 'http://localhost:5173',
  credentials: true
}));
Issue 4: Module Not Found
bash# Reinstall dependencies
rm -rf node_modules
npm install
Issue 5: TypeScript Errors
bash# Check types
npm run type-check

# Install missing types
npm install --save-dev @types/[package-name]

✅ Final Verification Script
bash#!/bin/bash
echo "🔍 Verifying AI SDR Platform Setup..."

# Check environment
if [ -f .env ]; then
    echo "✅ .env file exists"
else
    echo "❌ .env file missing"
    exit 1
fi

# Check dependencies
if [ -d node_modules ]; then
    echo "✅ Dependencies installed"
else
    echo "❌ Run npm install"
    exit 1
fi

# Check database
if psql $DATABASE_URL -c "SELECT 1" > /dev/null 2>&1; then
    echo "✅ Database connected"
else
    echo "❌ Database connection failed"
    exit 1
fi

# Check backend
if curl -s http://localhost:3000/health > /dev/null 2>&1; then
    echo "✅ Backend running"
else
    echo "❌ Backend not running - start with: npm run dev:backend"
fi

# Check frontend
if curl -s http://localhost:5173 > /dev/null 2>&1; then
    echo "✅ Frontend running"
else
    echo "❌ Frontend not running - start with: npm run dev:frontend"
fi

echo ""
echo "🎉 Setup verification complete!"
