import { storage } from "../storage";
import type { InsertProspect } from "../../shared/schema";

export interface ApolloConfig {
  apiKey: string;
  baseUrl: string;
}

export interface ApolloSearchParams {
  keywords?: string[];
  industries?: string[];
  companySize?: string;
  location?: string;
  jobTitles?: string[];
  limit?: number;
}

export interface ApolloProspect {
  id: string;
  firstName: string;
  lastName: string;
  email: string;
  jobTitle: string;
  company: {
    name: string;
    domain: string;
    industry: string;
    size: string;
    location: string;
  };
  linkedinUrl?: string;
  phone?: string;
}

class ApolloIntegrationService {
  private config: ApolloConfig = {
    apiKey: process.env.APOLLO_API_KEY || '',
    baseUrl: 'https://api.apollo.io/api/v1'
  };

  async searchProspects(params: ApolloSearchParams): Promise<ApolloProspect[]> {
    // Get API key from database settings
    const apolloApiKey = await storage.getSetting('apollo_api_key');
    if (!apolloApiKey) {
      console.log('⚠️ Apollo.io API key not configured, using sample data');
      return this.generateSampleProspects(params.limit || 10);
    }
    
    this.config.apiKey = apolloApiKey.value;

    try {
      const searchPayload = {
        api_key: this.config.apiKey,
        q_keywords: params.keywords?.join(' '),
        person_titles: params.jobTitles,
        organization_industry_tag_ids: this.mapIndustries(params.industries),
        organization_num_employees_ranges: this.mapCompanySize(params.companySize),
        organization_locations: params.location ? [params.location] : undefined,
        page: 1,
        per_page: Math.min(params.limit || 25, 100)
      };

      const response = await fetch(`${this.config.baseUrl}/people/search`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Cache-Control': 'no-cache',
          'accept': 'application/json'
        },
        body: JSON.stringify(searchPayload)
      });

      if (!response.ok) {
        throw new Error(`Apollo API error: ${response.status} ${response.statusText}`);
      }

      const data = await response.json();
      return this.transformApolloResults(data.people || []);

    } catch (error) {
      console.error('Apollo.io API error:', error);
      console.log('📊 Falling back to sample prospect data');
      return this.generateSampleProspects(params.limit || 10);
    }
  }

  private transformApolloResults(apolloData: any[]): ApolloProspect[] {
    return apolloData.map(person => ({
      id: person.id,
      firstName: person.first_name,
      lastName: person.last_name,
      email: person.email,
      jobTitle: person.title,
      company: {
        name: person.organization?.name || 'Unknown Company',
        domain: person.organization?.website_url || '',
        industry: person.organization?.industry || 'Technology',
        size: person.organization?.estimated_num_employees || 'Unknown',
        location: person.organization?.city || 'Unknown'
      },
      linkedinUrl: person.linkedin_url,
      phone: person.sanitized_phone
    }));
  }

  private mapIndustries(industries?: string[]): string[] | undefined {
    if (!industries) return undefined;
    
    const industryMap: Record<string, string> = {
      'technology': 'technology',
      'saas': 'computer_software',
      'finance': 'financial_services',
      'healthcare': 'hospital_health_care',
      'marketing': 'marketing_and_advertising',
      'consulting': 'management_consulting'
    };

    return industries.map(industry => 
      industryMap[industry.toLowerCase()] || industry
    );
  }

  private mapCompanySize(size?: string): string[] | undefined {
    if (!size) return undefined;
    
    const sizeMap: Record<string, string> = {
      'startup': '1,10',
      'small': '11,50', 
      'medium': '51,200',
      'large': '201,1000',
      'enterprise': '1001,10000'
    };

    return [sizeMap[size.toLowerCase()] || '1,10000'];
  }

  async addProspectsToDatabase(prospects: ApolloProspect[]): Promise<number> {
    let addedCount = 0;

    for (const prospect of prospects) {
      try {
        // Check if prospect already exists
        const existingProspects = await storage.getProspects();
        const exists = existingProspects.some(p => 
          p.email.toLowerCase() === prospect.email.toLowerCase()
        );

        if (exists) {
          console.log(`⚠️ Prospect ${prospect.email} already exists, skipping`);
          continue;
        }

        // For now, use a simple company ID (will be enhanced when company management is added)
        const companyId = 1; // Default company for imported prospects

        // Add prospect
        const newProspect: InsertProspect = {
          firstName: prospect.firstName,
          lastName: prospect.lastName,
          email: prospect.email,
          jobTitle: prospect.jobTitle,
          companyId: companyId,
          linkedinUrl: prospect.linkedinUrl,
          phone: prospect.phone,
          status: 'new'
        };

        await storage.createProspect(newProspect);
        addedCount++;
        
        console.log(`✅ Added prospect: ${prospect.firstName} ${prospect.lastName} from ${prospect.company.name}`);

      } catch (error) {
        console.error(`❌ Failed to add prospect ${prospect.email}:`, error);
      }
    }

    return addedCount;
  }

  async getConnectionStatus(): Promise<{
    connected: boolean;
    creditsRemaining?: number;
    lastSync?: string;
    error?: string;
  }> {
    const apolloApiKey = await storage.getSetting('apollo_api_key');
    if (!apolloApiKey) {
      return {
        connected: false,
        error: 'Apollo.io API key not configured'
      };
    }

    try {
      const response = await fetch(`${this.config.baseUrl}/auth/health`, {
        headers: {
          'Content-Type': 'application/json',
          'Cache-Control': 'no-cache',
          'X-Api-Key': this.config.apiKey
        }
      });

      if (response.ok) {
        return {
          connected: true,
          creditsRemaining: 1000, // Would be returned by API
          lastSync: new Date().toISOString()
        };
      } else {
        return {
          connected: false,
          error: `API connection failed: ${response.status}`
        };
      }
    } catch (error) {
      return {
        connected: false,
        error: `Connection error: ${(error as Error).message}`
      };
    }
  }
}

export const apolloIntegrationService = new ApolloIntegrationService();
