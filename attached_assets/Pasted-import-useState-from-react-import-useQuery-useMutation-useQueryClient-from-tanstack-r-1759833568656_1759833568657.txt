import { useState } from "react";
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Input } from "@/components/ui/input";
import { 
  Table, 
  TableBody, 
  TableCell, 
  TableHead, 
  TableHeader, 
  TableRow 
} from "@/components/ui/table";
import { 
  DropdownMenu, 
  DropdownMenuContent, 
  DropdownMenuItem, 
  DropdownMenuTrigger 
} from "@/components/ui/dropdown-menu";
import { Search, MoreHorizontal, Edit, Trash2, Play, Pause } from "lucide-react";
import { formatDistanceToNow } from "date-fns";
import { apiRequest } from "@/lib/queryClient";
import { useToast } from "@/hooks/use-toast";
import { useAuth } from "@/hooks/useAuth";
import type { EmailSequence } from "@shared/schema";

export default function SequenceList() {
  const [searchQuery, setSearchQuery] = useState("");
  const { toast } = useToast();
  const queryClient = useQueryClient();
  const { user, isAuthenticated } = useAuth();

  const { data: sequences, isLoading } = useQuery({
    queryKey: ["/api/sequences", user?.id],
    enabled: isAuthenticated && !!user?.id,
    keepPreviousData: true,
    refetchOnWindowFocus: false,
  });

  const updateSequenceMutation = useMutation({
    mutationFn: async ({ id, data }: { id: number; data: Partial<EmailSequence> }) => {
      return await apiRequest(`/api/sequences/${id}`, "PUT", data);
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["/api/sequences", user?.id] });
      toast({
        title: "Success",
        description: "Sequence updated successfully",
      });
    },
    onError: () => {
      toast({
        title: "Error",
        description: "Failed to update sequence",
        variant: "destructive",
      });
    },
  });

  const deleteSequenceMutation = useMutation({
    mutationFn: async (id: number) => {
      return await apiRequest(`/api/sequences/${id}`, "DELETE");
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["/api/sequences", user?.id] });
      toast({
        title: "Success",
        description: "Sequence deleted successfully",
      });
    },
    onError: () => {
      toast({
        title: "Error",
        description: "Failed to delete sequence",
        variant: "destructive",
      });
    },
  });

  const filteredSequences = (Array.isArray(sequences) ? sequences : [])
    .filter((sequence: EmailSequence) =>
      sequence.name.toLowerCase().includes(searchQuery.toLowerCase()) ||
      sequence.description?.toLowerCase().includes(searchQuery.toLowerCase())
    )
    .sort((a, b) => new Date(b.createdAt || 0).getTime() - new Date(a.createdAt || 0).getTime());

  const toggleSequenceStatus = (id: number, isActive: boolean) => {
    updateSequenceMutation.mutate({ id, data: { isActive: !isActive } });
  };

  const deleteSequence = (id: number) => {
    if (confirm("Are you sure you want to delete this sequence?")) {
      deleteSequenceMutation.mutate(id);
    }
  };

  if (isLoading) {
    return (
      <Card>
        <CardHeader>
          <CardTitle>Email Sequences</CardTitle>
        </CardHeader>
        <CardContent>
          <div className="space-y-4">
            {[...Array(3)].map((_, i) => (
              <div key={i} className="h-16 bg-gray-200 rounded animate-pulse" />
            ))}
          </div>
        </CardContent>
      </Card>
    );
  }

  return (
    <Card>
      <CardHeader>
        <div className="flex items-center justify-between">
          <CardTitle>Email Sequences</CardTitle>
          <div className="relative">
            <Search className="absolute left-3 top-3 h-4 w-4 text-gray-400" />
            <Input
              placeholder="Search sequences..."
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
              className="pl-10 w-64"
            />
          </div>
        </div>
      </CardHeader>
      <CardContent>
        {filteredSequences.length === 0 ? (
          <div className="text-center py-8">
            <p className="text-gray-500">
              {searchQuery ? "No sequences found matching your search" : "No sequences found"}
            </p>
          </div>
        ) : (
          <div className="overflow-x-auto">
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead>Name</TableHead>
                  <TableHead>Status</TableHead>
                  <TableHead>Emails</TableHead>
                  <TableHead>Response Rate</TableHead>
                  <TableHead>Created</TableHead>
                  <TableHead>Actions</TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {filteredSequences.map((sequence: EmailSequence) => (
                  <TableRow key={sequence.id}>
                    <TableCell>
                      <div>
                        <div className="font-medium">{sequence.name}</div>
                        <div className="text-sm text-gray-500">{sequence.description}</div>
                      </div>
                    </TableCell>
                    <TableCell>
                      <Badge variant={sequence.isActive ? "default" : "secondary"}>
                        {sequence.isActive ? "Active" : "Inactive"}
                      </Badge>
                    </TableCell>
                    <TableCell>
                      {sequence.emailCount || 0} emails
                    </TableCell>
                    <TableCell>
                      <div className="flex items-center space-x-2">
                        <span className={`font-medium ${
                          Number(sequence.responseRate || 0) > 20 ? 'text-green-600' : 
                          Number(sequence.responseRate || 0) > 15 ? 'text-yellow-600' : 
                          'text-gray-600'
                        }`}>
                          {sequence.responseRate || 0}%
                        </span>
                        <span className="text-sm text-gray-500">
                          ({sequence.totalResponses}/{sequence.totalSent})
                        </span>
                      </div>
                    </TableCell>
                    <TableCell>
                      {formatDistanceToNow(new Date(sequence.createdAt!), { addSuffix: true })}
                    </TableCell>
                    <TableCell>
                      <DropdownMenu>
                        <DropdownMenuTrigger asChild>
                          <Button variant="ghost" size="sm">
                            <MoreHorizontal className="w-4 h-4" />
                          </Button>
                        </DropdownMenuTrigger>
                        <DropdownMenuContent align="end">
                          <DropdownMenuItem onClick={() => toggleSequenceStatus(sequence.id, sequence.isActive || false)}>
                            {sequence.isActive ? (
                              <>
                                <Pause className="w-4 h-4 mr-2" />
                                Deactivate
                              </>
                            ) : (
                              <>
                                <Play className="w-4 h-4 mr-2" />
                                Activate
                              </>
                            )}
                          </DropdownMenuItem>
                          <DropdownMenuItem>
                            <Edit className="w-4 h-4 mr-2" />
                            Edit
                          </DropdownMenuItem>
                          <DropdownMenuItem 
                            onClick={() => deleteSequence(sequence.id)}
                            className="text-red-600"
                          >
                            <Trash2 className="w-4 h-4 mr-2" />
                            Delete
                          </DropdownMenuItem>
                        </DropdownMenuContent>
                      </DropdownMenu>
                    </TableCell>
                  </TableRow>
                ))}
              </TableBody>
            </Table>
          </div>
        )}
      </CardContent>
    </Card>
  );
}
