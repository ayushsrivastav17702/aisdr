Complete Enhanced Component Example:
jsximport { useState } from 'react';
import { Play, Pause, Edit2, Copy, Trash2, MoreVertical } from 'lucide-react';

export default function SequencesPage() {
  const [sequences, setSequences] = useState([]);
  const [view, setView] = useState('grid'); // 'grid' or 'list'
  const [filter, setFilter] = useState('all');
  const [searchQuery, setSearchQuery] = useState('');

  return (
    <div className="sequences-page">
      {/* Header */}
      <div className="page-header">
        <div>
          <h1 className="text-3xl font-bold">Sequences</h1>
          <p className="text-gray-600">Create and manage email sequences</p>
        </div>

        <button className="btn-primary">
          + New Sequence
        </button>
      </div>

      {/* Toolbar */}
      <div className="toolbar">
        <div className="search-box">
          <input
            type="text"
            placeholder="Search sequences..."
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
          />
        </div>

        <div className="toolbar-actions">
          {/* Filter */}
          <select 
            value={filter} 
            onChange={(e) => setFilter(e.target.value)}
            className="filter-select"
          >
            <option value="all">All Sequences</option>
            <option value="draft">Draft</option>
            <option value="active">Active</option>
            <option value="paused">Paused</option>
            <option value="completed">Completed</option>
          </select>

          {/* Sort */}
          <select className="sort-select">
            <option>Last Modified</option>
            <option>Name</option>
            <option>Performance</option>
          </select>

          {/* View Toggle */}
          <div className="view-toggle">
            <button 
              className={view === 'grid' ? 'active' : ''}
              onClick={() => setView('grid')}
            >
              Grid
            </button>
            <button 
              className={view === 'list' ? 'active' : ''}
              onClick={() => setView('list')}
            >
              List
            </button>
          </div>
        </div>
      </div>

      {/* Sequences Grid */}
      <div className={`sequences-${view}`}>
        {sequences.map((seq) => (
          <SequenceCard key={seq.id} sequence={seq} />
        ))}
      </div>
    </div>
  );
}

function SequenceCard({ sequence }) {
  return (
    <div className="sequence-card">
      {/* Header with actions */}
      <div className="card-header">
        <div className="title-row">
          <h3>{sequence.name}</h3>
          <div className="badges">
            <StatusBadge status={sequence.status} />
            {sequence.isActive && (
              <span className="active-indicator">●</span>
            )}
          </div>
        </div>

        <div className="quick-actions">
          <button title="Activate">
            <Play size={16} />
          </button>
          <button title="Edit">
            <Edit2 size={16} />
          </button>
          <button title="Clone">
            <Copy size={16} />
          </button>
          <button title="More">
            <MoreVertical size={16} />
          </button>
        </div>
      </div>

      {/* Description */}
      <p className="description">{sequence.description || 'No description'}</p>

      {/* Metrics */}
      <div className="metrics-grid">
        <div className="metric">
          <span className="metric-value">{sequence.prospects}</span>
          <span className="metric-label">Prospects</span>
        </div>
        <div className="metric">
          <span className="metric-value">{sequence.emailCount}</span>
          <span className="metric-label">Emails</span>
        </div>
        <div className="metric">
          <span className="metric-value">{sequence.openRate}%</span>
          <span className="metric-label">Open Rate</span>
        </div>
        <div className="metric">
          <span className="metric-value">{sequence.replyRate}%</span>
          <span className="metric-label">Reply Rate</span>
        </div>
      </div>

      {/* Progress (if active) */}
      {sequence.isActive && (
        <div className="progress-section">
          <div className="progress-bar">
            <div 
              className="progress-fill" 
              style={{ width: `${sequence.progress}%` }}
            />
          </div>
          <span className="progress-text">
            {sequence.contacted} of {sequence.prospects} contacted
          </span>
        </div>
      )}

      {/* Footer */}
      <div className="card-footer">
        <span className="timestamp">
          Modified {sequence.lastModified}
        </span>
        <button className="btn-view-details">
          View Details →
        </button>
      </div>
    </div>
  );
}
