Database Tables:
sqlCREATE TABLE email_domains (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    domain VARCHAR(255) NOT NULL,
    status VARCHAR(50), -- verified, pending, failed
    dns_verified BOOLEAN DEFAULT false,
    spf_record TEXT,
    dkim_record TEXT,
    dmarc_record TEXT,
    health_score INTEGER DEFAULT 100,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE mailboxes (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    email_address VARCHAR(255) NOT NULL,
    domain_id UUID REFERENCES email_domains(id),
    status VARCHAR(50), -- active, warming, suspended
    daily_limit INTEGER DEFAULT 50,
    sent_today INTEGER DEFAULT 0,
    health_score INTEGER DEFAULT 100,
    warmup_stage INTEGER DEFAULT 1,
    last_sent TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE email_queue (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    campaign_id UUID REFERENCES campaigns(id),
    prospect_id UUID REFERENCES prospects(id),
    mailbox_id UUID REFERENCES mailboxes(id),
    subject VARCHAR(255),
    body TEXT,
    status VARCHAR(50), -- pending, sent, failed, bounced
    scheduled_at TIMESTAMP,
    sent_at TIMESTAMP,
    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
