app.post('/api/sequences/:id/steps', async (req, res) => {
  const { steps } = req.body;
  
  await Sequence.findByIdAndUpdate(req.params.id, {
    steps: steps,
    stepCount: steps.length
  });
  
  res.json({ success: true });
});

// POST /api/sequences/:id/prospects
// Add prospects to sequence
app.post('/api/sequences/:id/prospects', async (req, res) => {
  const { prospectIds } = req.body;
  
  await SequenceProspect.insertMany(
    prospectIds.map(prospectId => ({
      sequenceId: req.params.id,
      prospectId,
      status: 'pending',
      currentStep: 0
    }))
  );
  
  await Sequence.findByIdAndUpdate(req.params.id, {
    $inc: { prospectCount: prospectIds.length }
  });
  
  res.json({ success: true, added: prospectIds.length });
});

// POST /api/sequences/:id/activate
// Activate sequence to start sending emails
app.post('/api/sequences/:id/activate', async (req, res) => {
  const sequence = await Sequence.findById(req.params.id);
  
  // Validation
  if (sequence.stepCount === 0) {
    return res.status(400).json({ error: 'Add email steps first' });
  }
  
  if (sequence.prospectCount === 0) {
    return res.status(400).json({ error: 'Add prospects first' });
  }
  
  // Activate
  await Sequence.findByIdAndUpdate(req.params.id, {
    status: 'active',
    activatedAt: new Date()
  });
  
  // Queue first emails for all prospects
  await queueInitialEmails(req.params.id);
  
  res.json({ success: true });
});

// POST /api/sequences/:id/pause
// Pause active sequence
app.post('/api/sequences/:id/pause', async (req, res) => {
  await Sequence.findByIdAndUpdate(req.params.id, {
    status: 'paused',
    pausedAt: new Date()
  });
  
  res.json({ success: true });
});
5. Email Sending Logic
javascript// Background job that runs every minute
async function processSequenceEmails() {
  // Find all prospects who should receive emails now
  const dueProspects = await SequenceProspect.find({
    status: 'active',
    nextEmailAt: { $lte: new Date() }
  }).populate('sequenceId prospectId');
  
  for (const sp of dueProspects) {
    const sequence = sp.sequenceId;
    const prospect = sp.prospectId;
    const currentStep = sequence.steps[sp.currentStep];
    
    // Send email
    await sendEmail({
      to: prospect.email,
      subject: currentStep.subject,
      body: personalizeEmail(currentStep.body, prospect)
    });
    
    // Update prospect status
    const nextStep = sp.currentStep + 1;
    
    if (nextStep >= sequence.steps.length) {
      // Sequence completed
      await SequenceProspect.updateOne(
        { _id: sp._id },
        { 
          status: 'completed',
          currentStep: nextStep
        }
      );
    } else {
      // Schedule next email
      const nextStepData = sequence.steps[nextStep];
      const nextEmailAt = new Date();
      nextEmailAt.setDate(nextEmailAt.getDate() + nextStepData.delayDays);
      
      await SequenceProspect.updateOne(
        { _id: sp._id },
        {
          currentStep: nextStep,
          nextEmailAt: nextEmailAt
        }
      );
    }
  }
}
