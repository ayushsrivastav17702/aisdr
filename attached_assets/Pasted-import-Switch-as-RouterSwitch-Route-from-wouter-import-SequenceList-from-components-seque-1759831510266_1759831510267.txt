                    <div className="space-y-2">
                      <Label>Industries</Label>
                      <Select onValueChange={(value) => {
                        setTargetingConfig(prev => ({
                          ...prev,
                          industries: prev.industries.includes(value)

                        <SelectTrigger>
                          <SelectValue placeholder="Select industries" />
                        </SelectTrigger>

                            <Badge key={location} variant="secondary" className="text-xs">
                              {location}

                          ))}

                    </div>
                  </div>

                  {/* Contact Filters */}
                  <div className="space-y-4">

                    <div className="space-y-2">
                      <Label>Job Titles</Label>

                          jobTitles: prev.jobTitles.includes(value)

                          <SelectValue placeholder="Select job titles" />

                    </span>
                  </div>
                  
                  <div className="flex space-x-2">
                    <Button 
                      variant="outline"
                      onClick={() => {
                        // Load saved config
                        if (savedConfigs.length > 0) {
                          const firstConfig = savedConfigs[0] as any;
                          setTargetingConfig(firstConfig);
                        }
                      }}
                    >
                      Load Saved
                    </Button>
                    <Button

                {/* Saved Configurations */}
                {savedConfigs.length > 0 && (

            </CardContent>
          </Card>
        </TabsContent>
      </Tabs>

      {/* Template Manager Dialog */}
      {showTemplateManager && (
        <EmailTemplateManager 
          onClose={() => setShowTemplateManager(false)}
        />
      )}
    </div>
  );
