// API helper
const apiRequest = async (url: string, method = 'GET', data?: any) => {
  const options: RequestInit = {
    method,
    headers: { 'Content-Type': 'application/json' }
  };
  if (data) options.body = JSON.stringify(data);
  
  const response = await fetch(url, options);
  if (!response.ok) throw new Error(`API Error: ${response.statusText}`);
  return response.json();
};

// ============================================
// SEQUENCE STEPS TAB
// ============================================
function SequenceStepsTab({ sequenceId }: { sequenceId: string }) {
  const [showAddModal, setShowAddModal] = useState(false);
  const [editingStep, setEditingStep] = useState<any>(null);
  const { toast } = useToast();
  const queryClient = useQueryClient();

  const { data: steps = [], isLoading } = useQuery({
    queryKey: ['/api/sequences', sequenceId, 'steps'],
    queryFn: () => apiRequest(`/api/sequences/${sequenceId}`).then(data => data.steps || []),
  });

  const addStepMutation = useMutation({
    mutationFn: (data: any) => apiRequest(`/api/sequences/${sequenceId}/steps`, 'POST', data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['/api/sequences', sequenceId] });
      setShowAddModal(false);
      toast({ title: "Success", description: "Email step added successfully" });
    },
  });

  return (
    <div className="space-y-4">
      <Card>
        <CardHeader>
          <div className="flex items-center justify-between">
            <div>
              <CardTitle>Email Sequence Steps</CardTitle>
              <CardDescription>Create and manage your email sequence flow</CardDescription>
            </div>
            <Button onClick={() => setShowAddModal(true)}>
              <Plus className="w-4 h-4 mr-2" />
              Add Step
            </Button>
          </div>
        </CardHeader>
        <CardContent>
          {isLoading ? (
            <div className="text-center py-8">
              <RefreshCw className="w-8 h-8 animate-spin mx-auto text-gray-400" />
              <p className="text-gray-500 mt-2">Loading steps...</p>
            </div>
          ) : steps.length === 0 ? (
            <div className="text-center py-12">
              <Mail className="w-16 h-16 text-gray-300 mx-auto mb-4" />
              <h3 className="text-lg font-semibold mb-2">No email steps yet</h3>
              <p className="text-gray-600 mb-4">Add your first email step to start the sequence</p>
              <Button onClick={() => setShowAddModal(true)}>
                <Plus className="w-4 h-4 mr-2" />
                Create First Step
              </Button>
            </div>
          ) : (
            <div className="space-y-4">
              {steps.map((step: any, index: number) => (
                <Card key={step.id} className="border-l-4 border-l-blue-500">
                  <CardContent className="pt-6">
                    <div className="flex items-start justify-between">
                      <div className="flex-1">
                        <div className="flex items-center gap-2 mb-2">
                          <Badge>Step {index + 1}</Badge>
                          {step.delayDays > 0 && (
                            <Badge variant="outline">
                              <Clock className="w-3 h-3 mr-1" />
                              Wait {step.delayDays} day{step.delayDays > 1 ? 's' : ''}
                            </Badge>
                          )}
                          {step.aiGenerated && (
                            <Badge variant="secondary">
                              <Sparkles className="w-3 h-3 mr-1" />
                              AI Generated
                            </Badge>
                          )}
                        </div>
                        <h4 className="font-semibold text-lg">{step.subject}</h4>
                        <p className="text-gray-600 text-sm mt-2 line-clamp-2">{step.body}</p>
                      </div>
                      <div className="flex items-center gap-2">
                        <Button variant="outline" size="sm" onClick={() => setEditingStep(step)}>
                          <Edit3 className="w-4 h-4" />
                        </Button>
                        <Button variant="outline" size="sm" className="text-red-600">
                          <Trash2 className="w-4 h-4" />
                        </Button>
                      </div>
                    </div>
                  </CardContent>
                </Card>
              ))}
            </div>
          )}
        </CardContent>
      </Card>

      {/* Add/Edit Step Modal */}
      <AddStepModal
        open={showAddModal || !!editingStep}
        onClose={() => {
          setShowAddModal(false);
          setEditingStep(null);
        }}
        onSave={(data) => addStepMutation.mutate(data)}
        step={editingStep}
        isPending={addStepMutation.isPending}
      />
    </div>
  );
}

// ============================================
// ADD STEP MODAL
// ============================================
function AddStepModal({ open, onClose, onSave, step, isPending }: any) {
  const [formData, setFormData] = useState({
    subject: step?.subject || '',
    body: step?.body || '',
    stepOrder: step?.stepOrder || 1,
    delayDays: step?.delayDays || 0,
  });

  useEffect(() => {
    if (step) {
      setFormData({
        subject: step.subject,
        body: step.body,
        stepOrder: step.stepOrder,
        delayDays: step.delayDays,
      });
    }
  }, [step]);

  if (!open) return null;

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle>{step ? 'Edit Email Step' : 'Add Email Step'}</DialogTitle>
        </DialogHeader>
        <div className="space-y-4">
          <div className="grid grid-cols-2 gap-4">
            <div>
              <Label>Step Order</Label>
              <Input
                type="number"
                value={formData.stepOrder}
                onChange={(e) => setFormData({ ...formData, stepOrder: parseInt(e.target.value) })}
              />
            </div>
            <div>
              <Label>Delay (Days)</Label>
              <Input
                type="number"
                value={formData.delayDays}
                onChange={(e) => setFormData({ ...formData, delayDays: parseInt(e.target.value) })}
              />
            </div>
          </div>
          <div>
            <Label>Subject Line</Label>
            <Input
              value={formData.subject}
              onChange={(e) => setFormData({ ...formData, subject: e.target.value })}
              placeholder="Enter email subject..."
            />
          </div>
          <div>
            <Label>Email Body</Label>
            <Textarea
              value={formData.body}
              onChange={(e) => setFormData({ ...formData, body: e.target.value })}
              placeholder="Enter email content..."
              rows={10}
            />
            <p className="text-xs text-gray-500 mt-1">
              Use variables: {'{firstName}'}, {'{lastName}'}, {'{company}'}, {'{jobTitle}'}
            </p>
          </div>
          <div className="flex gap-3 pt-4 border-t">
            <Button variant="outline" onClick={onClose} className="flex-1">Cancel</Button>
            <Button onClick={() => onSave(formData)} disabled={isPending} className="flex-1">
              {isPending ? 'Saving...' : 'Save Step'}
            </Button>
          </div>
        </div>
      </DialogContent>
    </Dialog>
  );
}

// ============================================
// LINKEDIN PERSONALIZATION MODAL
// ============================================
function LinkedInPersonalizationModal({ open, onClose, prospect, sequenceId }: any) {
  const [step, setStep] = useState(1);
  const [linkedInData, setLinkedInData] = useState({
    profileText: '',
    recentPosts: [''],
  });
  const [result, setResult] = useState<any>(null);
  const { toast } = useToast();

  const personalizeMutation = useMutation({
    mutationFn: (data: any) => apiRequest('/api/personalization/manual-linkedin', 'POST', data),
    onSuccess: (data) => {
      setResult(data);
      setStep(3);
      toast({ title: "Success", description: "Personalization complete!" });
    },
  });

  const handleGenerate = () => {
    if (!prospect?.id) {
      toast({ title: "Error", description: "No prospect selected", variant: "destructive" });
      return;
    }
    setStep(2);
    personalizeMutation.mutate({
      prospectId: prospect.id,
      linkedInData,
    });
  };

  if (!open) return null;

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent className="max-w-4xl max-h-[90vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle>AI Personalization with LinkedIn Data</DialogTitle>
          {prospect && (
            <p className="text-sm text-gray-500">
              Prospect: {prospect.firstName} {prospect.lastName} - {prospect.company}
            </p>
          )}
        </DialogHeader>

        <div className="space-y-4">
          {step === 1 && (
            <>
              <Card className="bg-blue-50 border-blue-200">
                <CardContent className="pt-6">
                  <div className="flex items-start gap-3">
                    <Sparkles className="w-5 h-5 text-blue-600 mt-0.5" />
                    <div>
                      <h3 className="font-semibold text-blue-900 mb-1">How it works</h3>
                      <p className="text-sm text-blue-800">
                        Paste LinkedIn profile data below. Our AI analyzes it using GPT-4 to generate highly personalized emails with 85-95% personalization scores.
                      </p>
                    </div>
                  </div>
                </CardContent>
              </Card>

              <div>
                <Label>LinkedIn Profile / About Section</Label>
                <Textarea
                  value={linkedInData.profileText}
                  onChange={(e) => setLinkedInData({ ...linkedInData, profileText: e.target.value })}
                  placeholder="Paste their LinkedIn About section here..."
                  rows={6}
                />
              </div>

              <div>
                <Label>Recent LinkedIn Posts (Optional)</Label>
                <Textarea
                  value={linkedInData.recentPosts[0]}
                  onChange={(e) => {
                    const newPosts = [...linkedInData.recentPosts];
                    newPosts[0] = e.target.value;
                    setLinkedInData({ ...linkedInData, recentPosts: newPosts });
                  }}
                  placeholder="Paste a recent post they made..."
                  rows={4}
                />
              </div>

              <div className="flex gap-3 pt-4 border-t">
                <Button variant="outline" onClick={onClose} className="flex-1">Cancel</Button>
                <Button
                  onClick={handleGenerate}
                  disabled={!linkedInData.profileText && !linkedInData.recentPosts[0]}
                  className="flex-1 bg-gradient-to-r from-blue-600 to-purple-600"
                >
                  <Sparkles className="w-4 h-4 mr-2" />
                  Generate AI Personalization
                </Button>
              </div>
            </>
          )}

          {step === 2 && (
            <div className="text-center py-12">
              <div className="inline-block animate-spin rounded-full h-16 w-16 border-4 border-blue-600 border-t-transparent mb-4"></div>
              <h3 className="text-xl font-semibold mb-2">Analyzing with OpenAI...</h3>
              <p className="text-gray-600">Creating personalized email content</p>
            </div>
          )}

          {step === 3 && result && (
            <>
              <Card className="bg-green-50 border-green-200">
                <CardContent className="pt-6">
                  <div className="flex items-center gap-2">
                    <CheckCircle className="w-5 h-5 text-green-600" />
                    <span className="font-medium">
                      Personalization Complete! Score: <span className="text-green-700">{result.personalizationScore}%</span>
                    </span>
                  </div>
                </CardContent>
              </Card>

              <div>
                <h3 className="font-semibold mb-3">Generated Email</h3>
                <div className="space-y-4">
                  <div>
                    <Label className="text-sm text-gray-700 mb-2">Subject Line</Label>
                    <div className="p-3 bg-gray-50 rounded border">{result.email?.subject}</div>
                  </div>
                  <div>
                    <Label className="text-sm text-gray-700 mb-2">Email Body</Label>
                    <div className="p-3 bg-gray-50 rounded border whitespace-pre-wrap">
                      {result.email?.body}
                    </div>
                  </div>
                </div>
              </div>

              <Button onClick={onClose} className="w-full">Done</Button>
            </>
          )}
        </div>
      </DialogContent>
    </Dialog>
  );
}
