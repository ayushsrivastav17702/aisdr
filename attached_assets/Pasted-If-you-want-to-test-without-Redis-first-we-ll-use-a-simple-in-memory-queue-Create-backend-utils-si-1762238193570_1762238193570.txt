If you want to test without Redis first, we'll use a simple in-memory queue.

javascript// Simple in-memory queue (for testing without Redis)
class SimpleQueue {
  constructor() {
    this.jobs = [];
    this.processing = false;
  }

  async add(jobName, data, options = {}) {
    const job = {
      id: Date.now(),
      name: jobName,
      data,
      options,
      status: 'pending'
    };
    
    this.jobs.push(job);
    console.log(`📥 Job added: ${jobName}`, data);
    
    // Start processing if not already
    if (!this.processing) {
      this.processJobs();
    }
    
    return job;
  }

  async processJobs() {
    this.processing = true;
    
    while (this.jobs.length > 0) {
      const job = this.jobs.shift();
      console.log(`⚙️ Processing job: ${job.name}`);
      
      try {
        // Import and run the handler
        if (job.name === 'automation') {
          const automationService = await import('../services/automationService.js');
          await automationService.default.processAutomation(
            job.data.automationRunId,
            job.data.sequenceId,
            job.data.prospectCount,
            job.data.aiPersonalization,
            job.data.apolloFilters
          );
        }
        
        job.status = 'completed';
        console.log(`✅ Job completed: ${job.name}`);
        
      } catch (error) {
        job.status = 'failed';
        console.error(`❌ Job failed: ${job.name}`, error);
      }
    }
    
    this.processing = false;
  }
}

export const automationQueue = new SimpleQueue();
