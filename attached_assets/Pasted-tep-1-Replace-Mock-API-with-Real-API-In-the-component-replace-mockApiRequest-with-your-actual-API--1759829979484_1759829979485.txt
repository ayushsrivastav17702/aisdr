Step 3: Add Missing Backend Routes
Add this to your backend/src/routes/sequences.ts:
typescript// Get replies for sequence
router.get('/sequences/:id/replies', async (req, res) => {
  const sequenceId = parseInt(req.params.id);
  const replies = await db
    .select({
      id: emailReplies.id,
      replyContent: emailReplies.replyContent,
      sentiment: emailReplies.sentiment,
      receivedAt: emailReplies.receivedAt,
      prospect: prospects
    })
    .from(emailReplies)
    .leftJoin(prospects, eq(emailReplies.prospectId, prospects.id))
    .leftJoin(emails, eq(emailReplies.emailId, emails.id))
    .where(eq(emails.sequenceId, sequenceId));

  res.json({ total: replies.length, replies });
});
