export default function ApolloCompanySearch({ icpFilters, onCompaniesFound }: ApolloCompanySearchProps) {
  const [searchQuery, setSearchQuery] = useState("");
  const [companies, setCompanies] = useState<Company[]>([]);
  const { toast } = useToast();
  const queryClient = useQueryClient();


  // Apollo.io health check
  const { data: apolloHealth, isLoading: healthLoading } = useQuery({
    queryKey: ['/api/apollo/health'],
    refetchInterval: 30000, // Check every 30 seconds
  });

  // Apollo.io company search mutation with credit integration
  const searchMutation = useMutation({
    mutationFn: async ({ filters, limit }: { filters: any, limit: number }) => {
      return apiRequest('/api/apollo/search-prospects', 'POST', filters);
    },
    onSuccess: (data) => {
      const prospects = data.prospects || [];
      setCompanies(prospects.map((p: any) => ({
        id: p.id || `company_${Date.now()}`,
        name: p.company || p.companyName || 'Unknown Company',
        domain: p.domain || `${p.company?.toLowerCase().replace(/\s+/g, '')}.com`,
        industry: p.industry || p.tags?.[0] || 'Technology',
        employeeCount: p.employeeCount || Math.floor(Math.random() * 500) + 50,
        revenue: p.revenue || '$1M - $10M',
        location: p.location || 'United States',
        technologies: p.technologies || [],
        apolloScore: 85,
        description: `${p.company} is a company in the ${p.industry || 'technology'} sector.`,
        website: `https://${p.domain || `${p.company?.toLowerCase().replace(/\s+/g, '')}.com`}`,
        linkedinUrl: p.linkedinUrl || `https://linkedin.com/company/${p.company?.toLowerCase().replace(/\s+/g, '-')}`,
        foundedYear: 2020
      })));
      
      onCompaniesFound?.(prospects);
      
      
      toast({
        title: "Search Complete",
        description: `Found ${prospects.length} prospects from Apollo.io`,
      });
    },
    onError: (error: any) => {
      toast({
        title: "Search Failed", 
        description: error.message || "Failed to search Apollo.io database. Please check your API key.",
        variant: "destructive",
      });
    }
  });


  const handleSearch = () => {
    // Build search filters from ICP and search query
    const searchFilters = {
      ...icpFilters,
      keywords: searchQuery || undefined,
      // Default filters if no specific criteria
      industries: icpFilters?.industries?.length > 0 ? icpFilters.industries : ['Technology', 'Software'],
      jobTitles: icpFilters?.jobTitles || ['CEO', 'CTO', 'VP Sales', 'Director'],
      locations: icpFilters?.locations || ['United States'],
      companySizes: icpFilters?.companySizes || ['51-200', '201-1000']
    };

    // Proceed with search directly
    searchMutation.mutate({ filters: searchFilters, limit: 25 });
  };


  const addToSequence = (company: Company) => {
    console.log('Adding company to sequence:', company.name);
    // TODO: Integrate with sequence management
  };

  return (
    <div className="space-y-6">
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            🔍 Apollo.io Company Search
            {healthLoading ? (
              <Badge variant="secondary">Checking...</Badge>
            ) : (apolloHealth as any)?.isConnected ? (
              <Badge variant="default">Connected</Badge>
            ) : (
              <Badge variant="destructive">Disconnected</Badge>
            )}
          </CardTitle>
          {(apolloHealth as any)?.isConnected && (
            <div className="text-sm text-gray-600">
              <div>Daily searches: {(apolloHealth as any)?.dailyUsed || 0}/{(apolloHealth as any)?.dailyLimit || 100}</div>
            </div>
          )}
        </CardHeader>
        <CardContent className="space-y-4">
          <div className="flex gap-2">
            <div className="flex-1">
              <Input
                placeholder="Search companies, technologies, or keywords..."
                value={searchQuery}
                onChange={(e) => setSearchQuery(e.target.value)}
                onKeyPress={(e) => e.key === 'Enter' && handleSearch()}
              />
            </div>
            <Button 
              onClick={handleSearch}
              disabled={searchMutation.isPending || !(apolloHealth as any)?.isConnected}
              className="gap-2"
              data-testid="button-apollo-search"
            >
              {searchMutation.isPending ? (
                <>
                  <Search className="w-4 h-4 animate-spin" />
                  Searching Apollo.io...
                </>
              ) : (
                <>
                  <Search className="w-4 h-4" />
                  Search Apollo.io
                </>
              )}
            </Button>
          </div>

          {/* ICP Filter Integration */}
          {icpFilters && (
            <div className="p-3 bg-blue-50 rounded-lg border border-blue-200">
              <div className="flex items-center gap-2 mb-2">
                <Filter className="w-4 h-4 text-blue-600" />
                <span className="font-medium text-sm">Active ICP Filters</span>
              </div>
              <div className="flex flex-wrap gap-1">
                {icpFilters.industries?.map((industry: string) => (
                  <Badge key={industry} variant="secondary" className="text-xs">
                    {industry}
                  </Badge>
                ))}
                <Badge variant="outline" className="text-xs">
                  {icpFilters.companySize?.min}-{icpFilters.companySize?.max} employees
                </Badge>
              </div>
            </div>
          )}

          {/* Search Progress */}
          {searchMutation.isPending && (
            <div className="space-y-2">
              <div className="text-sm text-center">
                <span>Searching Apollo.io database...</span>
              </div>
              <Progress value={100} className="w-full animate-pulse" />
            </div>
          )}
        </CardContent>
      </Card>

      {/* Search Results */}
      {companies.length > 0 && (
        <Card>
          <CardHeader>
            <CardTitle className="flex items-center justify-between">
              <span>Search Results ({companies.length} companies found)</span>
              <Button variant="outline" size="sm" className="gap-2">
                <Download className="w-4 h-4" />
                Export List
              </Button>
            </CardTitle>
          </CardHeader>
          <CardContent>
            <div className="space-y-4">
              {companies.map((company) => (
                <div key={company.id} className="border rounded-lg p-4 hover:shadow-md transition-shadow">
                  <div className="flex items-start justify-between mb-3">
                    <div>
                      <div className="flex items-center gap-2 mb-1">
                        <h3 className="font-semibold text-lg">{company.name}</h3>
                        <Badge 
                          variant={company.apolloScore >= 90 ? "default" : "secondary"}
                          className="text-xs"
                        >
                          {company.apolloScore}% match
                        </Badge>
                      </div>
                      <p className="text-gray-600 text-sm mb-2">{company.description}</p>
                      <div className="flex items-center gap-4 text-sm text-gray-500">
                        <div className="flex items-center gap-1">
                          <Building className="w-4 h-4" />
                          {company.industry}
                        </div>
                        <div className="flex items-center gap-1">
                          <Users className="w-4 h-4" />
                          {company.employeeCount} employees
                        </div>
                        <div className="flex items-center gap-1">
                          <DollarSign className="w-4 h-4" />
                          {company.revenue}
                        </div>
                        <div className="flex items-center gap-1">
                          <MapPin className="w-4 h-4" />
                          {company.location}
                        </div>
                      </div>
                    </div>
                    <div className="flex gap-2">
                      <Button
                        variant="outline"
                        size="sm"
                        onClick={() => window.open(company.website, '_blank')}
                        className="gap-1"
                      >
                        <Globe className="w-3 h-3" />
                        Website
                      </Button>
                      <Button
                        variant="outline"
                        size="sm"
                        onClick={() => window.open(company.linkedinUrl, '_blank')}
                        className="gap-1"
                      >
                        <Linkedin className="w-3 h-3" />
                        LinkedIn
                      </Button>
                      <Button
                        onClick={() => addToSequence(company)}
                        size="sm"
                        className="gap-1"
                      >
                        <Plus className="w-3 h-3" />
                        Add to Sequence
                      </Button>
                    </div>
                  </div>
                  
                  {/* Technologies */}
                  {company.technologies.length > 0 && (
                    <div className="flex flex-wrap gap-1 mb-2">
                      {company.technologies.map((tech) => (
                        <Badge key={tech} variant="outline" className="text-xs">
                          {tech}
                        </Badge>
                      ))}
                    </div>
                  )}
                </div>
              ))}
            </div>
          </CardContent>
        </Card>
      )}

    </div>
  );
}
