import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { useState } from "react";
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { Separator } from "@/components/ui/separator";
import { ScrollArea } from "@/components/ui/scroll-area";
import { Dialog, DialogContent, DialogDescription, DialogHeader, DialogTitle } from "@/components/ui/dialog";
import { Textarea } from "@/components/ui/textarea";
import { Label } from "@/components/ui/label";
import { useToast } from "@/hooks/use-toast";
import { Mail, CheckCircle, AlertCircle, RefreshCw, User, Building, Calendar, Reply, MessageSquare, Send, ExternalLink } from "lucide-react";
import { format } from "date-fns";

interface SequenceReply {
  id: number;
  prospectId: number;
  subject: string;
  content: string;
  sentAt: string;
  responseReceived: boolean;
  responseContent: string | null;
  responseAt: string | null;
  prospect: {
    id: number;
    firstName: string;
    lastName: string;
    email: string;
    company: string;
  } | null;
}

interface SequenceRepliesTabProps {
  sequenceId: number;
  sequenceName: string;
}

export default function SequenceRepliesTab({ sequenceId, sequenceName }: SequenceRepliesTabProps) {
  const [selectedReply, setSelectedReply] = useState<SequenceReply | null>(null);
  const [showFollowupDialog, setShowFollowupDialog] = useState(false);
  const [showThreadDialog, setShowThreadDialog] = useState(false);
  const [followupEmail, setFollowupEmail] = useState("");
  const [currentReply, setCurrentReply] = useState<SequenceReply | null>(null);
  const [generatingFollowup, setGeneratingFollowup] = useState(false);
  
  const queryClient = useQueryClient();
  const { toast } = useToast();

  // Fetch email replies for this specific sequence
  const { data: repliesData, isLoading, refetch } = useQuery({
    queryKey: [`/api/sequences/${sequenceId}/replies`],
    refetchInterval: 30000, // Refresh every 30 seconds
    staleTime: 10000, // Consider data stale after 10 seconds
  });

  const replies = (repliesData as any)?.data || [];
  const totalReplies = replies.length;

  // Handle generating AI follow-up email
  const handleGenerateFollowup = async (reply: SequenceReply) => {
    setCurrentReply(reply);
    setGeneratingFollowup(true);
    
    // Start with intelligent fallback email based on the reply content
    const generateSmartFollowup = (reply: SequenceReply) => {
      const firstName = reply.prospect?.firstName || 'there';
      const responseText = reply.responseContent?.toLowerCase() || '';
      
      if (responseText.includes('connect') || responseText.includes('interested')) {
        return `Hi ${firstName},

Great to hear you're interested in connecting! 

I'd love to schedule a brief 15-minute call to discuss how Increff can help ${reply.prospect?.company || 'your company'} optimize inventory management and improve operational efficiency.

I have availability this week on:
- Tuesday 2-4 PM IST
- Wednesday 10 AM-12 PM IST  
- Thursday 3-5 PM IST

Which time works better for you? Or feel free to suggest an alternative time that fits your schedule.

Looking forward to our conversation!

Best regards,
Mayukh Ganguly
Founder & CEO, Increff AI SDR
Phone: +91 98765 43210
Email: mayukh@increff.com`;
      }
      
      return `Hi ${firstName},

Thank you for your response! I appreciate you taking the time to get back to me.

I'd love to schedule a brief call to discuss how we can support ${reply.prospect?.company || 'your company'}'s goals and answer any questions you might have.

When would be a good time for you this week?

Best regards,
Mayukh Ganguly
Founder & CEO, Increff AI SDR
Phone: +91 98765 43210
Email: mayukh@increff.com`;
    };

    // Use smart fallback immediately for faster response
    setFollowupEmail(generateSmartFollowup(reply));
    setShowFollowupDialog(true);
    setGeneratingFollowup(false);
  };

  // Handle viewing full email thread
  const handleViewFullThread = (reply: SequenceReply) => {
    setCurrentReply(reply);
    setShowThreadDialog(true);
  };

  // Send follow-up email
  const sendFollowupMutation = useMutation({
    mutationFn: async (emailData: { to: string; subject: string; content: string }) => {
      const response = await fetch('/api/email/send', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          to: emailData.to,
          subject: emailData.subject,
          html: emailData.content.replace(/\n/g, '<br>'),
          text: emailData.content
        })
      });
      if (!response.ok) throw new Error('Failed to send email');
      return response.json();
    },
    onSuccess: () => {
      toast({ title: "Success", description: "Follow-up email sent successfully!" });
      setShowFollowupDialog(false);
      setFollowupEmail("");
      setCurrentReply(null);
    },
    onError: (error) => {
      toast({ title: "Error", description: "Failed to send follow-up email", variant: "destructive" });
    }
  });

  if (isLoading) {
    return (
      <div className="flex items-center justify-center py-8">
        <RefreshCw className="w-6 h-6 animate-spin text-blue-600" />
        <span className="ml-2 text-gray-600">Loading replies...</span>
      </div>
    );
  }

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex justify-between items-center">
        <div>
          <h3 className="text-lg font-semibold flex items-center">
            <Reply className="w-5 h-5 mr-2 text-green-600" />
            Email Replies - {sequenceName}
          </h3>
          <p className="text-sm text-gray-600">
            {totalReplies} prospect{totalReplies !== 1 ? 's' : ''} replied to this sequence
          </p>
        </div>
        <Button
          onClick={() => refetch()}
          variant="outline"
          size="sm"
          data-testid="button-refresh-sequence-replies"
        >
          <RefreshCw className="w-4 h-4 mr-2" />
          Refresh
        </Button>
      </div>

      {/* Statistics Cards */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
        <Card>
          <CardHeader className="pb-2">
            <CardTitle className="text-sm font-medium text-gray-600">Total Replies</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold text-green-600">{totalReplies}</div>
          </CardContent>
        </Card>
        <Card>
          <CardHeader className="pb-2">
            <CardTitle className="text-sm font-medium text-blue-600">Response Rate</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold text-blue-600">
              {totalReplies > 0 ? '100%' : '0%'}
            </div>
          </CardContent>
        </Card>
        <Card>
          <CardHeader className="pb-2">
            <CardTitle className="text-sm font-medium text-purple-600">Latest Reply</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-sm text-purple-600">
              {replies.length > 0 
                ? format(new Date(replies[0].responseAt || replies[0].sentAt), 'MMM dd, yyyy')
                : 'No replies yet'
              }
            </div>
          </CardContent>
        </Card>
      </div>

      {/* Replies List */}
      {totalReplies === 0 ? (
        <Card>
          <CardContent className="text-center py-8">
            <MessageSquare className="w-12 h-12 mx-auto mb-4 text-gray-300" />
            <h3 className="text-lg font-medium text-gray-600 mb-2">No Replies Yet</h3>
            <p className="text-gray-500">
              When prospects reply to emails in this sequence, they'll appear here.
            </p>
          </CardContent>
        </Card>
      ) : (
        <div className="space-y-4">
          {replies.map((reply: SequenceReply) => (
            <Card 
              key={reply.id} 
              className={`cursor-pointer transition-colors hover:bg-gray-50 ${
                selectedReply?.id === reply.id ? 'ring-2 ring-blue-500' : ''
              }`}
              onClick={() => setSelectedReply(selectedReply?.id === reply.id ? null : reply)}
              data-testid={`reply-card-${reply.id}`}
            >
              <CardHeader className="pb-3">
                <div className="flex items-center justify-between">
                  <div className="flex items-center space-x-3">
                    <div className="w-10 h-10 bg-green-100 rounded-full flex items-center justify-center">
                      <User className="w-5 h-5 text-green-600" />
                    </div>
                    <div>
                      <div className="font-medium">
                        {reply.prospect?.firstName} {reply.prospect?.lastName}
                      </div>
                      <div className="text-sm text-gray-500 flex items-center">
                        <Building className="w-3 h-3 mr-1" />
                        {reply.prospect?.company}
                      </div>
                    </div>
                  </div>
                  <div className="text-right">
                    <Badge variant="outline" className="bg-green-50 text-green-700 border-green-200">
                      <CheckCircle className="w-3 h-3 mr-1" />
                      Replied
                    </Badge>
                    <div className="text-xs text-gray-500 mt-1 flex items-center">
                      <Calendar className="w-3 h-3 mr-1" />
                      {format(new Date(reply.responseAt || reply.sentAt), 'MMM dd, yyyy HH:mm')}
                    </div>
                  </div>
                </div>
              </CardHeader>

              <CardContent className="pt-0">
                <div className="space-y-3">
                  <div>
                    <div className="text-sm font-medium text-gray-700">Original Subject:</div>
                    <div className="text-sm text-gray-600">{reply.subject}</div>
                  </div>

                  {selectedReply?.id === reply.id && reply.responseContent && (
                    <div className="border-t pt-3 mt-3">
                      <div className="text-sm font-medium text-gray-700 mb-2">Prospect's Reply:</div>
                      <ScrollArea className="max-h-48 w-full">
                        <div className="text-sm text-gray-600 whitespace-pre-wrap bg-gray-50 p-3 rounded-lg">
                          {reply.responseContent}
                        </div>
                      </ScrollArea>
                    </div>
                  )}

                  {selectedReply?.id === reply.id && (
                    <div className="flex gap-2 pt-2">
                      <Button 
                        size="sm" 
                        variant="outline"
                        onClick={(e) => {
                          e.stopPropagation();
                          handleGenerateFollowup(reply);
                        }}
                        disabled={generatingFollowup}
                        data-testid={`button-generate-followup-${reply.id}`}
                      >
                        <Mail className="w-4 h-4 mr-2" />
                        {generatingFollowup ? 'Generating...' : 'Generate Follow-up'}
                      </Button>
                      <Button 
                        size="sm" 
                        variant="outline"
                        onClick={(e) => {
                          e.stopPropagation();
                          handleViewFullThread(reply);
                        }}
                        data-testid={`button-view-thread-${reply.id}`}
                      >
                        View Full Thread
                      </Button>
                    </div>
                  )}
                </div>
              </CardContent>
            </Card>
          ))}
        </div>
      )}

      {/* Follow-up Email Generation Dialog */}
      <Dialog open={showFollowupDialog} onOpenChange={setShowFollowupDialog}>
        <DialogContent className="max-w-2xl">
          <DialogHeader>
            <DialogTitle>Generate Follow-up Email</DialogTitle>
            <DialogDescription>
              AI-generated follow-up email for {currentReply?.prospect?.firstName} {currentReply?.prospect?.lastName}
            </DialogDescription>
          </DialogHeader>
          
          <div className="space-y-4">
            <div>
              <Label htmlFor="to">To:</Label>
              <div className="text-sm text-gray-600 p-2 bg-gray-50 rounded">
                {currentReply?.prospect?.email}
              </div>
            </div>
            
            <div>
              <Label htmlFor="subject">Subject:</Label>
              <div className="text-sm text-gray-600 p-2 bg-gray-50 rounded">
                Re: {currentReply?.subject}
              </div>
            </div>
            
            <div>
              <Label htmlFor="followupEmail">Email Content:</Label>
              <Textarea
                id="followupEmail"
                value={followupEmail}
                onChange={(e) => setFollowupEmail(e.target.value)}
                rows={10}
                placeholder="Your follow-up email content..."
                className="resize-none"
              />
            </div>
            
            <div className="flex justify-end gap-2">
              <Button 
                variant="outline" 
                onClick={() => setShowFollowupDialog(false)}
              >
                Cancel
              </Button>
              <Button 
                onClick={() => {
                  if (currentReply?.prospect?.email) {
                    sendFollowupMutation.mutate({
                      to: currentReply.prospect.email,
                      subject: `Re: ${currentReply.subject}`,
                      content: followupEmail
                    });
                  }
                }}
                disabled={sendFollowupMutation.isPending || !followupEmail.trim()}
              >
                <Send className="w-4 h-4 mr-2" />
                {sendFollowupMutation.isPending ? 'Sending...' : 'Send Email'}
              </Button>
            </div>
          </div>
        </DialogContent>
      </Dialog>

      {/* Full Thread View Dialog */}
      <Dialog open={showThreadDialog} onOpenChange={setShowThreadDialog}>
        <DialogContent className="max-w-3xl max-h-[80vh]">
          <DialogHeader>
            <DialogTitle>Email Thread</DialogTitle>
            <DialogDescription>
              Full conversation with {currentReply?.prospect?.firstName} {currentReply?.prospect?.lastName}
            </DialogDescription>
          </DialogHeader>
          
          <ScrollArea className="max-h-[60vh] w-full">
            <div className="space-y-4">
              {/* Original Email */}
              <div className="border-l-4 border-blue-500 pl-4">
                <div className="text-sm font-medium text-gray-700 mb-2">
                  Original Email (Sent {currentReply ? format(new Date(currentReply.sentAt), 'MMM dd, yyyy HH:mm') : ''})
                </div>
                <div className="text-sm font-medium mb-1">Subject: {currentReply?.subject}</div>
                <div className="text-sm text-gray-600 bg-blue-50 p-3 rounded-lg whitespace-pre-wrap">
                  {currentReply?.content ? 
                    currentReply.content.replace(/<[^>]*>/g, '').replace(/&nbsp;/g, ' ') : 
                    'Email content not available'
                  }
                </div>
              </div>

              {/* Reply */}
              {currentReply?.responseContent && (
                <div className="border-l-4 border-green-500 pl-4">
                  <div className="text-sm font-medium text-gray-700 mb-2">
                    Reply (Received {format(new Date(currentReply.responseAt || currentReply.sentAt), 'MMM dd, yyyy HH:mm')})
                  </div>
                  <div className="text-sm font-medium mb-1">From: {currentReply.prospect?.email}</div>
                  <div className="text-sm text-gray-600 bg-green-50 p-3 rounded-lg whitespace-pre-wrap">
                    {currentReply.responseContent}
                  </div>
                </div>
              )}
            </div>
          </ScrollArea>
          
          <div className="flex justify-end">
            <Button 
              variant="outline" 
              onClick={() => setShowThreadDialog(false)}
            >
              Close
            </Button>
          </div>
        </DialogContent>
      </Dialog>
    </div>
  );
}
