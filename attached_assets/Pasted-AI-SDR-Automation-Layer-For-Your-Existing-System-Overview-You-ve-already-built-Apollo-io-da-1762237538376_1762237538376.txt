  // Return immediately

    message: 'Automation started. Processing in background...'
  });
});
2. Process Automation (Background)
javascript// This runs asynchronously after API returns

async function processAutomation(automationRunId, prospectCount, aiPersonalization, sequenceId, apolloFilters) {
  try {
    // STEP 1: Fetch prospects from Apollo
    console.log(`[Automation ${automationRunId}] Fetching ${prospectCount} prospects from Apollo...`);
    
    const apolloProspects = await yourExistingApolloFunction(apolloFilters, prospectCount);
    
    if (!apolloProspects || apolloProspects.length === 0) {
      throw new Error('No prospects found');
    }

    console.log(`[Automation ${automationRunId}] Found ${apolloProspects.length} prospects`);

    // STEP 2: Save prospects to your existing prospects table
    const savedProspectIds = [];
    
    for (const prospect of apolloProspects) {
      const prospectId = await yourExistingSaveProspectFunction(prospect);
      
      // Link to automation run
      await db.query(
        'UPDATE prospects SET automation_run_id = $1 WHERE id = $2',
        [automationRunId, prospectId]
      );
      
      savedProspectIds.push(prospectId);
    }

    // Update automation run
    await db.query(
      'UPDATE automation_runs SET prospects_added = $1 WHERE id = $2',
      [savedProspectIds.length, automationRunId]
    );

    // STEP 3: For each prospect, enroll in sequence
    for (const prospectId of savedProspectIds) {
      await enrollProspectInSequence(prospectId, sequenceId, automationRunId, aiPersonalization);
    }

    // STEP 4: Mark automation as completed
    await db.query(
      `UPDATE automation_runs 
       SET status = 'completed', completed_at = NOW() 
       WHERE id = $1`,
      [automationRunId]
    );

    console.log(`[Automation ${automationRunId}] Completed successfully`);

  } catch (error) {
    console.error(`[Automation ${automationRunId}] Failed:`, error);
    
    await db.query(
      `UPDATE automation_runs SET status = 'failed' WHERE id = $1`,
      [automationRunId]
    );
  }
}

4. Schedule Email (Uses Your Existing Email Logic)
javascriptasync function scheduleEmail(prospectId, sequenceId, stepNumber, delayDays, aiPersonalization) {
  // Get prospect info
  const prospect = await db.query('SELECT * FROM prospects WHERE id = $1', [prospectId]);
  const prospectData = prospect.rows[0];

  // Calculate send time (your existing delay logic)
  const sendAt = new Date();
  sendAt.setDate(sendAt.getDate() + delayDays);
  sendAt.setHours(10, 0, 0, 0); // 10 AM

  // If AI personalization is enabled
  if (aiPersonalization) {
    // Call your EXISTING AI personalization function
    const { subject, body } = await yourExistingAIPersonalizationFunction(
      prospectData,
      sequenceId,
      stepNumber
    );

    // Save email to queue/schedule
    await yourExistingSendEmailFunction(prospectId, sequenceId, stepNumber, subject, body, sendAt);
  } else {
    // Use template from sequence
    const template = await getSequenceStepTemplate(sequenceId, stepNumber);
    
    await yourExistingSendEmailFunction(
      prospectId, 
      sequenceId, 
      stepNumber, 
      template.subject, 
      template.body, 
      sendAt
    );
  }
}

javascript// Add to your existing reply detection code:

function classifyReply(replyText) {
  const lowerText = replyText.toLowerCase();
  
  // Unsubscribe detection
  if (lowerText.includes('unsubscribe') || 
      lowerText.includes('stop emailing') ||
      lowerText.includes('remove me')) {
    return 'UNSUBSCRIBE';
  }
  
  // Out of office
  if (lowerText.includes('out of office') || 
      lowerText.includes('ooo') ||
      lowerText.includes('away from office')) {
    return 'OUT_OF_OFFICE';
  }
  
  // Negative
  if (lowerText.includes('not interested') || 
      lowerText.includes('no thanks')) {
    return 'NEGATIVE';
  }
  
  // Positive/Question
  return 'POSITIVE';
}

// When reply comes in:
const replyType = classifyReply(reply.body);

if (replyType === 'UNSUBSCRIBE' || replyType === 'NEGATIVE') {
  // STOP sequence immediately

  // DO NOT send AI follow-up
  return;
}

if (replyType === 'OUT_OF_OFFICE') {
  // Skip AI follow-up, but keep sequence active
  return;
}

// Only send AI follow-up for positive replies
if (replyType === 'POSITIVE') {

javascript// Add to your delay calculation:

import moment from 'moment-timezone';

function calculateSendTime(prospect, delayDays) {
  const timezone = prospect.timezone || 'America/New_York';
  
  let sendTime = moment.tz(timezone)
    .add(delayDays, 'days')
    .hour(10) // 10 AM local time
    .minute(0)
    .second(0);
  
  // Skip weekends
  if (sendTime.day() === 0) sendTime.add(1, 'day'); // Sunday → Monday
  if (sendTime.day() === 6) sendTime.add(2, 'days'); // Saturday → Monday
  
  return sendTime.toDate();
}
