  id: number;
  firstName: string;
  lastName: string;

  private async fetchContentManagementData(): Promise<any> {
    try {

      const { storage } = await import('../../storage');

      console.log(`✅ Found content items: ${content.length}`);
      
      return {
        caseStudies: content.filter((item: any) => item.type === 'case_study'),
        whitepapers: content.filter((item: any) => item.type === 'whitepaper'),
        blogPosts: content.filter((item: any) => item.type === 'blog_post'),
        templates: content.filter((item: any) => item.type === 'template'),

      };
    } catch (error) {
      console.error('❌ Failed to fetch content management data:', error);

  private extractIndustryFromContent(content: string): { industry: string; size: string; technologies: string[] } {
    const lowerContent = content.toLowerCase();
    
    // Industry detection
    let industry = "Technology";
    if (lowerContent.includes("retail") || lowerContent.includes("ecommerce") || lowerContent.includes("fashion")) {
      industry = "Retail/E-commerce";
    } else if (lowerContent.includes("finance") || lowerContent.includes("banking")) {
      industry = "Financial Services";
    } else if (lowerContent.includes("healthcare") || lowerContent.includes("medical")) {
      industry = "Healthcare";
    } else if (lowerContent.includes("manufacturing") || lowerContent.includes("supply chain")) {
      industry = "Manufacturing";
    }
    
    // Size estimation
    let size = "medium";
    if (lowerContent.includes("startup") || lowerContent.includes("founded 20")) {
      size = "startup";
    } else if (lowerContent.includes("enterprise") || lowerContent.includes("global")) {
      size = "enterprise";
    }
    
    // Technology stack detection
    const technologies: string[] = [];
    if (lowerContent.includes("cloud")) technologies.push("Cloud Computing");
    if (lowerContent.includes("ai") || lowerContent.includes("artificial intelligence")) technologies.push("AI/ML");
    if (lowerContent.includes("mobile")) technologies.push("Mobile Technology");
    if (lowerContent.includes("saas")) technologies.push("SaaS Platform");
    
    return { industry, size, technologies };
  }

  private buildEnrichedContext(prospect: ProspectData, linkedinData: LinkedInData | null, companyData: CompanyData | null, contentData: any): string {
    let context = `
PROSPECT INFORMATION:
- Name: ${prospect.firstName} ${prospect.lastName}
- Job Title: ${prospect.jobTitle}
- Company: ${prospect.company}
- Email: ${prospect.email}`;

    if (prospect.linkedinUrl) {
      context += `
- LinkedIn Profile: ${prospect.linkedinUrl}`;
    }

    if (prospect.websiteUrl) {
      context += `
- Company Website: ${prospect.websiteUrl}`;
    }

    if (linkedinData) {
      context += `

LINKEDIN PROFILE INSIGHTS:
- Professional Experience: ${linkedinData.experience.join(', ')}
- Skills: ${linkedinData.skills.join(', ')}
- Recent Activity: ${linkedinData.posts.join(', ')}
- Network Size: ${linkedinData.connections}+ connections`;
    }

    if (companyData) {
      context += `

COMPANY RESEARCH:
- Industry: ${companyData.industry}
- Company Description: ${companyData.description}
- Size: ${companyData.size}
- Technologies: ${companyData.technologies.join(', ')}`;
    }

    if (contentData.allContent.length > 0) {
      context += `

AVAILABLE CONTENT FOR PERSONALIZATION:
- Case Studies: ${contentData.caseStudies.map((cs: any) => cs.title).join(', ')}
- Whitepapers: ${contentData.whitepapers.map((wp: any) => wp.title).join(', ')}
- Blog Posts: ${contentData.blogPosts.map((bp: any) => bp.title).join(', ')}

CONTENT DETAILS:
${contentData.allContent.slice(0, 3).map((item: any) => `- ${item.title}: ${item.content?.substring(0, 100)}...`).join('\n')}`;
    }

    if (prospect.notes) {
      context += `

ADDITIONAL PROSPECT NOTES:
${prospect.notes}`;
    }

    return context;
  }

  private extractCompanyName(prospect: ProspectData): string {
    if (prospect.notes?.includes('Company:')) {
      return prospect.notes.split('Company:')[1].split(',')[0].trim();
    }
    return prospect.company || 'the company';
  }

    } else if (jobTitle.includes('sales')) {
      challenges = ["Pipeline generation", "Conversion rates", "Sales efficiency"];

    return {

        competitors: [],
