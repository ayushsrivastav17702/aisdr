import { pgTable, serial, text, integer, boolean, timestamp, jsonb, real } from 'drizzle-orm/pg-core';

// Prospects Table
export const prospects = pgTable('prospects', {
  id: serial('id').primaryKey(),
  firstName: text('first_name').notNull(),
  lastName: text('last_name').notNull(),
  email: text('email').notNull().unique(),
  company: text('company'),
  jobTitle: text('job_title'),
  industry: text('industry'),
  phone: text('phone'),
  linkedinUrl: text('linkedin_url'),
  websiteUrl: text('website_url'),
  leadScore: integer('lead_score').default(0),
  status: text('status').default('new'),
  personalizedAt: timestamp('personalized_at'),
  personalizationScore: integer('personalization_score'),
  lastContactedAt: timestamp('last_contacted_at'),
  createdAt: timestamp('created_at').notNull().defaultNow(),
  updatedAt: timestamp('updated_at').notNull().defaultNow()
});

// Sequences Table
export const sequences = pgTable('sequences', {
  id: serial('id').primaryKey(),
  name: text('name').notNull(),
  description: text('description'),
  type: text('type').notNull().default('outbound'),
  status: text('status').notNull().default('draft'),
  createdBy: integer('created_by'),
  aiPersonalizationEnabled: boolean('ai_personalization_enabled').default(false),
  aiPersonalizationJobId: integer('ai_personalization_job_id'),
  totalProspects: integer('total_prospects').default(0),
  activeProspects: integer('active_prospects').default(0),
  completedProspects: integer('completed_prospects').default(0),
  responseRate: real('response_rate').default(0),
  openRate: real('open_rate').default(0),
  clickRate: real('click_rate').default(0),
  settings: jsonb('settings'),
  createdAt: timestamp('created_at').notNull().defaultNow(),
  updatedAt: timestamp('updated_at').notNull().defaultNow()
});

// Sequence Steps Table
export const sequenceSteps = pgTable('sequence_steps', {
  id: serial('id').primaryKey(),
  sequenceId: integer('sequence_id').notNull().references(() => sequences.id, { onDelete: 'cascade' }),
  subject: text('subject').notNull(),
  body: text('body').notNull(),
  stepOrder: integer('step_order').notNull(),
  delayDays: integer('delay_days').notNull().default(0),
  stepType: text('step_type').notNull().default('email'),
  aiGenerated: boolean('ai_generated').default(false),
  variables: jsonb('variables'),
  createdAt: timestamp('created_at').notNull().defaultNow(),
  updatedAt: timestamp('updated_at').notNull().defaultNow()
});

// Sequence Prospects Table

  id: serial('id').primaryKey(),
  sequenceId: integer('sequence_id').notNull().references(() => sequences.id, { onDelete: 'cascade' }),
  prospectId: integer('prospect_id').notNull().references(() => prospects.id, { onDelete: 'cascade' }),
  currentStepId: integer('current_step_id').references(() => sequenceSteps.id),

// 7. SEQUENCE ROUTES (backend/src/routes/sequences.ts)
// ============================================

import { Router } from 'express';
import { db } from '../db';
import { sequences, sequenceSteps, sequenceProspects, prospects } from '../../../shared/schema';
import { eq, sql } from 'drizzle-orm';

const router = Router();

// Get all sequences
router.get('/sequences', async (req, res) => {
  try {
    const allSequences = await db.select().from(sequences);
    res.json(allSequences);
  } catch (error) {
    res.status(500).json({ error: 'Failed to fetch sequences' });
  }
});

// Get single sequence with steps
router.get('/sequences/:id', async (req, res) => {
  try {
    const sequenceId = parseInt(req.params.id);
    const [sequence] = await db.select().from(sequences).where(eq(sequences.id, sequenceId));
    if (!sequence) return res.status(404).json({ error: 'Not found' });
    
    const steps = await db.select().from(sequenceSteps)
      .where(eq(sequenceSteps.sequenceId, sequenceId))
      .orderBy(sequenceSteps.stepOrder);
    
    res.json({ ...sequence, steps });
  } catch (error) {
    res.status(500).json({ error: 'Failed to fetch sequence' });
  }
});

// Create sequence
router.post('/sequences', async (req, res) => {
  try {
    const { name, description, type } = req.body;
    const [newSequence] = await db.insert(sequences).values({
      name,
      description,
      type: type || 'outbound',
      status: 'draft',
      createdAt: new Date(),
      updatedAt: new Date()
    }).returning();
    res.json(newSequence);
  } catch (error) {
    res.status(500).json({ error: 'Failed to create sequence' });
  }
});

// Add step to sequence
router.post('/sequences/:id/steps', async (req, res) => {
  try {
    const sequenceId = parseInt(req.params.id);
    const { subject, body, stepOrder, delayDays } = req.body;
    
    const [newStep] = await db.insert(sequenceSteps).values({
      sequenceId,
      subject,
      body,
      stepOrder: stepOrder || 1,
      delayDays: delayDays || 0,
      stepType: 'email',
      createdAt: new Date()
    }).returning();
    
    res.json(newStep);
  } catch (error) {
    res.status(500).json({ error: 'Failed to add step' });
  }
});

// Get prospects in sequence
router.get('/sequences/:id/prospects', async (req, res) => {
  try {
    const sequenceId = parseInt(req.params.id);
    const enrolledProspects = await db
      .select({
        id: sequenceProspects.id,
        status: sequenceProspects.status,
        enrolledAt: sequenceProspects.enrolledAt,
        prospect: prospects
      })
      .from(sequenceProspects)
      .leftJoin(prospects, eq(sequenceProspects.prospectId, prospects.id))
      .where(eq(sequenceProspects.sequenceId, sequenceId));
    
    res.json({ total: enrolledProspects.length, prospects: enrolledProspects });
  } catch (error) {
    res.status(500).json({ error: 'Failed to fetch prospects' });
  }
});

// Add prospects to sequence
router.post('/sequences/:id/prospects', async (req, res) => {
  try {
    const sequenceId = parseInt(req.params.id);
    const { prospectIds } = req.body;
    
    const enrolled = [];
    for (const prospectId of prospectIds) {
      const [result] = await db.insert(sequenceProspects).values({
        sequenceId,
        prospectId,
        status: 'active',
        enrolledAt: new Date()
      }).returning();
      enrolled.push(result);
    }
    
    res.json({ message: `${enrolled.length} prospects enrolled`, enrolled });
  } catch (error) {
    res.status(500).json({ error: 'Failed to add prospects' });
  }
});

// Get email tracking stats
router.get('/sequences/:id/tracking', async (req, res) => {
  try {
    const sequenceId = parseInt(req.params.id);

    res.json({
      sent: 0,
      delivered: 0,
      opened: 0,
      replied: 0,
      deliveryRate: 0,
      openRate: 0,
      replyRate: 0
    });
  } catch (error) {
    res.status(500).json({ error: 'Failed to fetch tracking' });
  }
});

// 8. PERSONALIZATION ROUTES (backend/src/routes/personalization.ts)

import { Router } from 'express';
import { db } from '../db';
import { prospects, personalizationResults, contentLibrary } from '../../../shared/schema';
import { eq } from 'drizzle-orm';
import OpenAI from 'openai';

const router = Router();
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// Manual LinkedIn personalization
router.post('/personalization/manual-linkedin', async (req, res) => {
  try {
    const { prospectId, linkedInData } = req.body;
    
    const [prospect] = await db.select().from(prospects).where(eq(prospects.id, prospectId));
    if (!prospect) return res.status(404).json({ error: 'Prospect not found' });
    
    // Analyze with OpenAI
    const prompt = `Analyze LinkedIn data and generate personalized email:

Prospect: ${prospect.firstName} ${prospect.lastName}
Company: ${prospect.company}
Title: ${prospect.jobTitle}

LinkedIn Data:
Profile: ${linkedInData.profileText || 'N/A'}
Recent Posts: ${linkedInData.recentPosts?.join('\n') || 'None'}
Recent Comments: ${linkedInData.recentComments?.join('\n') || 'None'}

Provide JSON:
{
  "analysis": {
    "professionalFocus": [],
    "painPoints": [],
    "recentInterests": []
  },
  "email": {
    "subject": "",
    "body": ""
  },
  "personalizationScore": 0-100
}`;

    const completion = await openai.chat.completions.create({
      model: 'gpt-4',
      messages: [
        { role: 'system', content: 'You are a B2B sales email expert.' },
        { role: 'user', content: prompt }
      ],
      temperature: 0.7,
      max_tokens: 1500
    });

    const result = JSON.parse(completion.choices[0].message.content);
    
    // Save result
    await db.insert(personalizationResults).values({
      prospectId,
      personalizationScore: result.personalizationScore,
      insights: result.analysis,
      emailSuggestions: result.email,
      linkedinData: linkedInData,
      createdAt: new Date()
    });
    
    res.json({
      message: 'Personalization complete',
      linkedInAnalysis: result.analysis,
      email: result.email,
      personalizationScore: result.personalizationScore
    });
  } catch (error) {
    console.error('Personalization error:', error);
    res.status(500).json({ error: 'Failed to generate personalization' });
  }
});

export default router;

// 9. WEBHOOK ROUTES (backend/src/routes/webhooks.ts)
// ============================================

import { Router } from 'express';
import { db } from '../db';
import { emails, emailReplies, sequenceProspects } from '../../../shared/schema';
import { eq, sql } from 'drizzle-orm';
import OpenAI from 'openai';

const router = Router();
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// Email reply webhook
router.post('/email-reply', async (req, res) => {
  try {
    const { emailId, prospectId, from, body, receivedAt } = req.body;
    
    // Analyze sentiment with AI
    const sentimentPrompt = `Analyze email reply sentiment: "${body}"
Return JSON: {"sentiment": "positive|negative|neutral|interested", "summary": "brief summary", "nextAction": "recommendation"}`;
    
    const completion = await openai.chat.completions.create({
      model: 'gpt-4',
      messages: [
        { role: 'system', content: 'You are a sales email analyst.' },
        { role: 'user', content: sentimentPrompt }
      ],
      temperature: 0.3,
      max_tokens: 300
    });

    const analysis = JSON.parse(completion.choices[0].message.content);
    
    // Save reply
    await db.insert(emailReplies).values({
      emailId,
      prospectId,
      replyContent: body,
      sentiment: analysis.sentiment,
      aiSummary: analysis.summary,
      nextAction: analysis.nextAction,
      receivedAt: new Date(receivedAt || Date.now()),
      createdAt: new Date()
    });
    
    // Update email status
    await db.update(emails).set({ repliedAt: new Date(), status: 'replied' })
      .where(eq(emails.id, emailId));
    
    // Update prospect status
    await db.update(sequenceProspects).set({ 
      status: 'replied', 
      replies: sql`replies + 1` 
    }).where(eq(sequenceProspects.prospectId, prospectId));
    
    res.json({ message: 'Reply processed', sentiment: analysis.sentiment });
  } catch (error) {
    console.error('Webhook error:', error);
    res.status(500).json({ error: 'Failed to process reply' });
  }
});

// Email opened webhook
router.post('/email-opened', async (req, res) => {
  try {
    const { trackingId, timestamp } = req.body;
    const [email] = await db.select().from(emails).where(eq(emails.trackingId, trackingId));
    
    if (email && !email.openedAt) {
      await db.update(emails).set({ 
        openedAt: new Date(timestamp || Date.now()),
        status: 'opened'
      }).where(eq(emails.id, email.id));
    }
    
    res.json({ message: 'Open tracked' });
  } catch (error) {
    res.status(500).json({ error: 'Failed to track open' });
  }
});

// Tracking pixel endpoint
router.get('/pixel/:trackingId', async (req, res) => {
  const { trackingId } = req.params;
  
  // Track open asynchronously
  db.select().from(emails).where(eq(emails.trackingId, trackingId)).then(([email]) => {
    if (email && !email.openedAt) {
      db.update(emails).set({ openedAt: new Date(), status: 'opened' })
        .where(eq(emails.id, email.id));
    }
  });
  
  // Return 1x1 transparent pixel
  const pixel = Buffer.from('R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7', 'base64');
  res.writeHead(200, {
    'Content-Type': 'image/gif',
    'Content-Length': pixel.length,
    'Cache-Control': 'no-cache'
  });
  res.end(pixel);
});

export default router;

import React, { useState } from 'react';
import { Users, MessageSquare, Zap, BarChart3, Settings } from 'lucide-react';

export default function SequenceBuilder() {
  const [activeTab, setActiveTab] = useState('sequence');
  
  const tabs = [
    { id: 'sequence', label: 'Sequence', icon: null },
    { id: 'prospects', label: 'Prospects', icon: Users },
    { id: 'replies', label: 'Replies', icon: MessageSquare },
    { id: 'ai-followup', label: 'AI Follow-up', icon: Zap },
    { id: 'tracking', label: 'Email Tracking', icon: BarChart3 },
    { id: 'settings', label: 'Settings', icon: Settings }
  ];

  return (
    <div className="min-h-screen bg-gray-50">
      <div className="max-w-7xl mx-auto p-6">
        <h1 className="text-3xl font-bold mb-6">Sequence Builder</h1>
        
        {/* Tabs */}
        <div className="bg-white rounded-lg shadow-sm mb-6">
          <div className="border-b border-gray-200 flex">
            {tabs.map((tab) => {
              const Icon = tab.icon;
              return (
                <button
                  key={tab.id}
                  onClick={() => setActiveTab(tab.id)}
                  className={`px-6 py-4 text-sm font-medium border-b-2 ${
                    activeTab === tab.id
                      ? 'border-blue-600 text-blue-600'
                      : 'border-transparent text-gray-600'
                  }`}
                >
                  {Icon && <Icon className="w-4 h-4 inline mr-2" />}
                  {tab.label}
                </button>
              );
            })}
          </div>
        </div>

        {/* Content */}
        <div className="bg-white rounded-lg shadow-sm p-6">
          {activeTab === 'sequence' && <div>Sequence Steps Content</div>}
          {activeTab === 'prospects' && <div>Prospects Content</div>}
          {activeTab === 'replies' && <div>Replies Content</div>}
          {activeTab === 'ai-followup' && <div>AI Follow-up Content</div>}
          {activeTab === 'tracking' && <div>Email Tracking Content</div>}
          {activeTab === 'settings' && <div>Settings Content</div>}
        </div>
      </div>
    </div>
  );
}
