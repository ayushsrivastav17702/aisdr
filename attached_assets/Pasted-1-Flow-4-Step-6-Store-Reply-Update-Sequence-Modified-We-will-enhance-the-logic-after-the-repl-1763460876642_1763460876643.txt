2. New Service: SequenceStepService.cancelFutureSteps
This new function is what actually pauses the automated flow by removing or marking future scheduled jobs.

JavaScript

// NEW function within SequenceStepService.js

/**
 * Removes future email jobs from the queue for a specific prospect in a sequence.
 */
static async cancelFutureSteps(sequenceId, prospectId) {
  // Find all pending email queue items for this prospect/sequence
  const futureSteps = await db.select()
    .from(emailQueue)
    .where(and(
      eq(emailQueue.sequenceId, sequenceId),
      eq(emailQueue.prospectId, prospectId),
      eq(emailQueue.status, "pending") // Only cancel jobs that haven't been sent
    ));
    
  if (futureSteps.length > 0) {
    // Update the status of all found jobs to 'cancelled'
    await db.update(emailQueue)
      .set({ 
        status: "cancelled", 
        cancellationReason: "Prospect replied/unsubscribed" 
      })
      .where(and(
        eq(emailQueue.sequenceId, sequenceId),
        eq(emailQueue.prospectId, prospectId),
        eq(emailQueue.status, "pending")
      ));
    
    console.log(`✅ Cancelled ${futureSteps.length} future emails for prospect ${prospectId}`);
  }
}
