import cron from 'node-cron';
import { db } from '../../db';
import { emailSequences } from '@shared/schema';
import { eq } from 'drizzle-orm';

class AIPersonalizationScheduler {
  private isRunning = false;

  /**
   * Start the AI personalization scheduler
   * Runs daily at 9 AM to process prospects for all enabled sequences
   */
  startScheduler() {
    // Run every day at 9 AM
    cron.schedule('0 9 * * *', async () => {
      if (this.isRunning) {
        console.log('⚠️  AI personalization job already running, skipping...');
        return;
      }

      this.isRunning = true;
      console.log('🤖 Starting daily AI personalization job...');
      
      try {
        // Get all sequences with AI personalization enabled
        const activeSequences = await db
          .select()
          .from(emailSequences)
          .where(eq(emailSequences.aiPersonalizationEnabled, true));
        
        console.log(`Found ${activeSequences.length} sequences with AI personalization enabled`);
        
        // Process each sequence
        for (const sequence of activeSequences) {
          try {
            console.log(`🔄 Processing sequence ${sequence.id}: "${sequence.name}"`);
            
            const response = await fetch(`http://0.0.0.0:${process.env.PORT || 5000}/api/sequences/${sequence.id}/process-ai-batch`, {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' }
            });
            
            if (!response.ok) {
              throw new Error(`HTTP ${response.status}: ${response.statusText}`);
            }
            
            const result = await response.json();
            console.log(`✅ Processed sequence ${sequence.id}: ${result.succeeded}/${result.processed} succeeded`);
            
          } catch (error) {
            console.error(`❌ Failed to process sequence ${sequence.id}:`, error instanceof Error ? error.message : String(error));
          }
        }
        
        console.log('✅ Daily AI personalization job completed');
        
      } catch (error) {
        console.error('❌ Daily AI personalization job failed:', error);
      } finally {
        this.isRunning = false;
      }
    });
    
    console.log('📅 AI Personalization Scheduler initialized - runs daily at 9 AM');
  }

  /**
   * Manually trigger batch processing for a specific sequence (for testing)
   */
  async triggerManualBatch(sequenceId: number): Promise<void> {
    console.log(`🧪 Manually triggering AI personalization for sequence ${sequenceId}`);
    
    try {
      const response = await fetch(`http://0.0.0.0:${process.env.PORT || 5000}/api/sequences/${sequenceId}/process-ai-batch`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' }
      });
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }
      
      const result = await response.json();
      console.log(`✅ Manual batch completed: ${result.succeeded}/${result.processed} succeeded`);
      
    } catch (error) {
      console.error('❌ Manual batch failed:', error);
      throw error;
    }
  }
}

export const aiPersonalizationScheduler = new AIPersonalizationScheduler();
