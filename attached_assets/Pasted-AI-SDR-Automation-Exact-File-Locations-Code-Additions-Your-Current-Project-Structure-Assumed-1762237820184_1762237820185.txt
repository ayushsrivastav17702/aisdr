javascript// ============================================
// AUTOMATION API ROUTES
// ============================================

import express from 'express';
import { body, validationResult } from 'express-validator';
import db from '../config/database.js'; // Your existing DB connection
import automationService from '../services/automationService.js'; // We'll create this next

const router = express.Router();

// ==========================================
// POST /api/automation/start
// Start a new automation run
// ==========================================
router.post('/start', [
  body('sequenceId').isInt().withMessage('Valid sequence ID required'),
  body('prospectCount').isInt({ min: 1, max: 500 }).withMessage('Prospect count must be 1-500'),
  body('aiPersonalization').isBoolean(),
  body('apolloFilters').isObject()
], async (req, res) => {
  // Validate input
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }

  const { sequenceId, prospectCount, aiPersonalization, apolloFilters } = req.body;
  const userId = req.user?.id || 1; // Get from your auth middleware

  try {
    // 1. Verify sequence exists
    const sequenceCheck = await db.query(
      'SELECT id, name, status FROM sequences WHERE id = $1',
      [sequenceId]
    );

    if (sequenceCheck.rows.length === 0) {
      return res.status(404).json({ error: 'Sequence not found' });
    }

    const sequence = sequenceCheck.rows[0];

    if (sequence.status !== 'active') {
      return res.status(400).json({ error: 'Sequence must be active to run automation' });
    }

    // 2. Create automation run record
    const result = await db.query(
      `INSERT INTO automation_runs 
       (sequence_id, prospect_count, ai_personalization_enabled, apollo_filters, created_by, status)
       VALUES ($1, $2, $3, $4, $5, 'running')
       RETURNING id`,
      [sequenceId, prospectCount, aiPersonalization, JSON.stringify(apolloFilters), userId]
    );

    const automationRunId = result.rows[0].id;

    // 3. Start processing in background (don't await)
    automationService.processAutomation(
      automationRunId,
      sequenceId,
      prospectCount,
      aiPersonalization,
      apolloFilters
    ).catch(err => {
      console.error(`Automation ${automationRunId} failed:`, err);
    });

    // 4. Return immediately
    res.json({
      success: true,
      automationRunId,
      message: `Automation started. Processing ${prospectCount} prospects in background...`
    });

  } catch (error) {
    console.error('Error starting automation:', error);
    res.status(500).json({ error: 'Failed to start automation' });
  }
});

import db from '../config/database.js';
import apolloService from './apolloService.js';      // Your existing
import aiService from './aiService.js';              // Your existing

class AutomationService {
  
  // ==========================================
  // Main automation processing function
  // ==========================================
  async processAutomation(automationRunId, sequenceId, prospectCount, aiPersonalization, apolloFilters) {
    console.log(`[Automation ${automationRunId}] Starting...`);

    try {
      // =====================================
      // STEP 1: Fetch prospects from Apollo
      // =====================================
      console.log(`[Automation ${automationRunId}] Fetching ${prospectCount} prospects from Apollo...`);
      
      // Call YOUR existing Apollo function
      const apolloProspects = await apolloService.searchProspects({
        ...apolloFilters,
        limit: prospectCount
      });

      if (!apolloProspects || apolloProspects.length === 0) {
        throw new Error('No prospects found from Apollo with given filters');
      }

      console.log(`[Automation ${automationRunId}] Found ${apolloProspects.length} prospects`);

      // =====================================
      // STEP 2: Save prospects to database
      // =====================================
      const savedProspectIds = [];

      for (const apolloProspect of apolloProspects) {
        try {
          // Call YOUR existing save prospect function
          // Adjust this based on your actual function name
          const prospectId = await this.saveOrUpdateProspect(apolloProspect, automationRunId);
          
          savedProspectIds.push(prospectId);

        } catch (error) {
          console.error(`Error saving prospect ${apolloProspect.email}:`, error.message);
          // Continue with other prospects
        }
      }

      // Update automation run
      await db.query(
        'UPDATE automation_runs SET prospects_added = $1 WHERE id = $2',
        [savedProspectIds.length, automationRunId]
      );

      console.log(`[Automation ${automationRunId}] Saved ${savedProspectIds.length} prospects`);

      // =====================================
      // STEP 3: Enroll prospects in sequence
      // =====================================
      for (const prospectId of savedProspectIds) {
        try {
          await this.enrollProspectInSequence(
            prospectId,
            sequenceId,
            automationRunId,
            aiPersonalization
          );
        } catch (error) {
          console.error(`Error enrolling prospect ${prospectId}:`, error.message);
        }
      }

      // =====================================
      // STEP 4: Mark automation as completed
      // =====================================
      await db.query(
        `UPDATE automation_runs 
         SET status = 'completed', completed_at = NOW() 
         WHERE id = $1`,
        [automationRunId]
      );

      console.log(`[Automation ${automationRunId}] Completed successfully!`);

    } catch (error) {
      console.error(`[Automation ${automationRunId}] Failed:`, error);

      // Mark as failed
      await db.query(
        `UPDATE automation_runs 
         SET status = 'failed', errors = $1 
         WHERE id = $2`,
        [error.message, automationRunId]
      );
    }
  }

  // ==========================================
  // Save or update prospect in database
  // ==========================================
  async saveOrUpdateProspect(apolloProspect, automationRunId) {
    // Check if prospect already exists
    const existing = await db.query(
      'SELECT id FROM prospects WHERE email = $1',
      [apolloProspect.email]
    );

    let prospectId;

    if (existing.rows.length > 0) {
      // Update existing prospect
      prospectId = existing.rows[0].id;

      await db.query(
        `UPDATE prospects SET 
         first_name = $1,
         last_name = $2,
         company = $3,
         job_title = $4,
         linkedin_url = $5,
         location = $6,
         automation_run_id = $7,
         updated_at = NOW()
         WHERE id = $8`,
        [
          apolloProspect.first_name,
          apolloProspect.last_name,
          apolloProspect.organization?.name || apolloProspect.company,
          apolloProspect.title || apolloProspect.job_title,
          apolloProspect.linkedin_url,
          apolloProspect.city || apolloProspect.location,
          automationRunId,
          prospectId
        ]
      );

    } else {
      // Insert new prospect
      const result = await db.query(
        `INSERT INTO prospects 
         (email, first_name, last_name, company, job_title, linkedin_url, location, automation_run_id)
         VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
         RETURNING id`,
        [
          apolloProspect.email,
          apolloProspect.first_name,
          apolloProspect.last_name,
          apolloProspect.organization?.name || apolloProspect.company,
          apolloProspect.title || apolloProspect.job_title,
          apolloProspect.linkedin_url,
          apolloProspect.city || apolloProspect.location,
          automationRunId
        ]
      );

      prospectId = result.rows[0].id;
    }

    return prospectId;
  }

  // ==========================================
  // Enroll prospect in sequence
  // ==========================================
  async enrollProspectInSequence(prospectId, sequenceId, automationRunId, aiPersonalization) {
    // Get sequence steps
    const stepsResult = await db.query(
      'SELECT * FROM sequence_steps WHERE sequence_id = $1 ORDER BY step_number',
      [sequenceId]
    );

    if (stepsResult.rows.length === 0) {
      throw new Error('Sequence has no steps');
    }

    // Check if prospect already enrolled (prevent duplicates)
    const existingEnrollment = await db.query(
      'SELECT id FROM prospect_sequences WHERE prospect_id = $1 AND sequence_id = $2',
      [prospectId, sequenceId]
    );

    if (existingEnrollment.rows.length > 0) {
      console.log(`Prospect ${prospectId} already enrolled in sequence ${sequenceId}`);
      return;
    }

    // Create prospect_sequence record
    await db.query(
      `INSERT INTO prospect_sequences 
       (prospect_id, sequence_id, automation_run_id, current_step, status)
       VALUES ($1, $2, $3, 1, 'active')`,
      [prospectId, sequenceId, automationRunId]
    );

    // Schedule first email
    const firstStep = stepsResult.rows[0];
    
    // Call YOUR existing email scheduling function
    // This should use your existing delay logic, AI generation, etc.
    await emailService.scheduleEmail({
      prospectId,
      sequenceId,
      stepNumber: 1,
      delayDays: firstStep.delay_days || 0,
      aiPersonalization
    });

    console.log(`Enrolled prospect ${prospectId} in sequence ${sequenceId}`);
  }
}

export default new AutomationService();

function generateUnsubscribeToken(prospectId) {
  const crypto = require('crypto');
  const secret = process.env.UNSUBSCRIBE_SECRET || 'change-this-secret';
  
  const data = `${prospectId}-${Date.now()}`;
  return crypto.createHmac('sha256', secret).update(data).digest('hex') + '.' + prospectId;
}

function verifyUnsubscribeToken(token) {
  try {
    const [hash, prospectId] = token.split('.');
    return parseInt(prospectId);
  } catch {
    return null;
  }
}

function classifyReply(replyText) {
  const text = replyText.toLowerCase();
  
  // 1. Check for unsubscribe requests
  const unsubscribeKeywords = [
    'unsubscribe',
    'stop emailing',
    'remove me',
    'take me off',
    'stop sending',
    'do not contact',
    'leave me alone'
  ];
  
  if (unsubscribeKeywords.some(keyword => text.includes(keyword))) {
    return 'UNSUBSCRIBE';
  }
  
  // 2. Check for out of office
  const oooKeywords = [
    'out of office',
    'out of the office',
    'ooo',
    'away from office',
    'automatic reply',
    'auto-reply',
    'on vacation'
  ];
  
  if (oooKeywords.some(keyword => text.includes(keyword))) {
    return 'OUT_OF_OFFICE';
  }
  
  // 3. Check for negative responses
  const negativeKeywords = [
    'not interested',
    'no thank',
    'don\'t contact',
    'already have',
    'not a good fit',
    'not the right time',
    'pass on this'
  ];
  
  if (negativeKeywords.some(keyword => text.includes(keyword))) {
    return 'NEGATIVE';
  }
  
  // 4. Check for questions (positive engagement)
  if (text.includes('?') || text.includes('tell me more') || text.includes('interested')) {
    return 'QUESTION';
  }
  
  // 5. Default to positive
  return 'POSITIVE';
}

async function handleReply(reply, prospectId, sequenceId) {
  // Classify the reply
  const classification = classifyReply(reply.body);
  
  console.log(`Reply from prospect ${prospectId} classified as: ${classification}`);
  
  // Save reply to database
  await db.query(
    `INSERT INTO replies 
     (prospect_id, subject, body, classification, received_at)
     VALUES ($1, $2, $3, $4, NOW())`,
    [prospectId, reply.subject, reply.body, classification]
  );
  
  // Handle based on classification
  switch (classification) {
    case 'UNSUBSCRIBE':
      // STOP ALL sequences for this prospect
      await db.query(
        `UPDATE prospect_sequences 
         SET status = 'unsubscribed', updated_at = NOW() 
         WHERE prospect_id = $1`,
        [prospectId]
      );
      
      await db.query(
        `INSERT INTO unsubscribes (prospect_id, email, reason)
         VALUES ($1, (SELECT email FROM prospects WHERE id = $1), 'Email request')`,
        [prospectId]
      );
      
      console.log(`Prospect ${prospectId} unsubscribed via email`);
      return; // Don't send AI follow-up
      
    case 'NEGATIVE':
      // Pause sequence but don't unsubscribe
      await db.query(
        `UPDATE prospect_sequences 
         SET status = 'opted_out', updated_at = NOW() 
         WHERE prospect_id = $1 AND sequence_id = $2`,
        [prospectId, sequenceId]
      );
      
      console.log(`Prospect ${prospectId} opted out (negative response)`);
      return; // Don't send AI follow-up
      
    case 'OUT_OF_OFFICE':
      // Don't send AI follow-up, but keep sequence active
      console.log(`Prospect ${prospectId} sent OOO, skipping AI response`);
      return;
      
    case 'QUESTION':
    case 'POSITIVE':
      // STOP sequence (they replied!)
      await db.query(
        `UPDATE prospect_sequences 
         SET status = 'replied', updated_at = NOW() 
         WHERE prospect_id = $1 AND sequence_id = $2`,
        [prospectId, sequenceId]
      );
      
      // NOW call YOUR existing AI follow-up function
      await yourExistingAIFollowUpFunction(prospectId, reply.body);
      break;
  }
}

          prospectCount: config.prospectCount,
          aiPersonalization: config.aiPersonalization,
          apolloFilters: config.apolloFilters
        })
      });

      const data = await response.json();

      if (data.success) {
        alert(`Automation started! ID: ${data.automationRunId}\n\nProcessing ${config.prospectCount} prospects in background.`);
        onSuccess();
      } else {
        setError(data.error || 'Failed to start automation');
      }
    } catch (err) {
      console.error('Automation error:', err);
      setError('Failed to start automation. Please try again.');
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <div className="modal-overlay" onClick={onClose}>
      <div className="modal-content" onClick={(e) => e.stopPropagation()}>
        <div className="modal-header">
          <h2>🤖 Run Automation</h2>
          <button className="close-btn" onClick={onClose}>×</button>
        </div>

        <form onSubmit={handleSubmit}>
          <div className="form-section">
            <h3>Sequence: {sequence.name}</h3>
            <p className="text-muted">{sequence.steps} steps configured</p>
          </div>

          <div className="form-group">
            <label>
              Number of Prospects *
              <input
                type="number"
                min="1"
                max="500"
                value={config.prospectCount}
                onChange={(e) => setConfig({...config, prospectCount: parseInt(e.target.value)})}
                required
